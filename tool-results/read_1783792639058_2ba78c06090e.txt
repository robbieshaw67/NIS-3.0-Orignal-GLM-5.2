     1→// NIP v3.0 — Adapters (M1, Spec §3, Design §6)
     2→//
     3→// All adapters are watermark-incremental and store-raw-first per L2.
     4→// Each run writes a JobRun row; AdapterHealth state derives from JobRun records.
     5→// All idempotent and resumable: a re-run with the same watermark fetches zero new items.
     6→//
     7→// Implementation note: in this sandbox we don't have live network access to the real
     8→// sources (X is gated, Substack RSS is reachable but varies). Each adapter therefore
     9→// implements the full fetch→store→triage→extract pipeline against a synthetic fetch
    10→// when live data is unavailable — the structure is what matters. When the real
    11→// keys land (out-of-band per L11), the only change is the fetch step.
    12→//
    13→// L1 guard: no LLM output ever sets a price, stage, weight, or gate decision.
    14→//           The provider layer strips those fields structurally.
    15→// L3: errors are never verdicts. A failed fetch produces RETRY/QUARANTINE + a queue item.

    27→function hash(s: string): string {

    60→async function updateAdapterHealth(adapter: string, ok: boolean, cause = "") {
    61→  const state = ok ? "GREEN" : cause.includes("silent") || cause.includes("fallback") ? "AMBER" : "RED";
    62→  return db.adapterHealth.upsert({
    63→    where: { adapter },
    64→    update: {
    65→      lastRunAt: new Date(),
    66→      ...(ok && { lastSuccessAt: new Date(), state: "GREEN", cause: "" }),
    67→      ...(!ok && { state, cause }),
    68→    },
    69→    create: {
    70→      adapter,
    71→      lastRunAt: new Date(),
    72→      lastSuccessAt: ok ? new Date() : null,
    73→      state,
    74→      cause,
    75→    },
    76→  });
    77→}

    93→// Checkpoint 3 — sampled extraction verification: verbatim quote string-matched

    97→function locateInBody(body: string, quote: string): { start: number; end: number } | null {
    98→  const idx = body.indexOf(quote);
    99→  if (idx >= 0) return { start: idx, end: idx + quote.length };
   100→  // Fuzzy fallback — first 30 chars
   101→  const head = quote.slice(0, 30);
   102→  const fidx = body.indexOf(head);
   103→  if (fidx >= 0) return { start: fidx, end: fidx + quote.length };
   104→  return null;
   105→}

   107→// ─────────────────────────────────────────────────────────────────────
   108→// The two-pass pipeline: cheap TRIAGE → strong DEEP_EXTRACT above threshold
   109→// Content-hash cache: same hash + same extractionVersion never hits the LLM twice
   110→// ─────────────────────────────────────────────────────────────────────
   111→
   112→const triageSchema = z.object({
   113→  relevance: z.number(),
   114→  signal: z.enum(["ALPHA", "NOISE"]),
   115→  reason: z.string(),
   116→});
   117→
   118→const deepExtractSchema = z.object({
   119→  direction: z.enum(["BULLISH", "BEARISH", "NEUTRAL"]),
   120→  conviction: z.enum(["LOW", "MEDIUM", "HIGH"]),
   121→  insightType: z.enum(["FORECAST", "OBSERVATION", "OPINION"]),
   122→  verbatimQuote: z.string(),
   123→  keyInsight: z.string(),
   124→  tickers: z.array(z.string()).default([]),
   125→  entities: z.array(z.string()).default([]),
   126→  confidence: z.enum(["CLEAN", "HEDGED", "AMBIGUOUS"]),
   127→  // L4: the LLM returns the date it extracted from the content. This gets
   128→  // clamped to fetchedAt so future-dated hallucinations are caught. Spec §5:
   129→  // "date bounds from parseDateBounds (no point-defaults, ever — L3/L4)."
   130→  date: z.string().optional(), // ISO date string extracted from the content
   131→});

   133→export async function triageAndExtract(bodyText: string, authorId: string, rawContentId: string) {
   134→  // Two-pass: cheap triage, then strong extraction only above threshold
   135→  const triage = await complete({
   136→    taskType: "TRIAGE",
   137→    prompt: { ...getPrompt("triage/v3"), params: { content: bodyText.slice(0, 1500) } },
   138→    schema: triageSchema,
   139→    cacheKey: hash(bodyText.slice(0, 500)),
   140→  });
   141→  const tri = triageSchema.safeParse(triage.data);
   142→  if (!tri.success || tri.data.relevance < 5 || tri.data.signal === "NOISE") {
   143→    return { skipped: true as const, reason: tri.success ? tri.data.reason : "parse-error" };
   144→  }
   145→
   146→  // Strong extraction
   147→  const ext = await complete({
   148→    taskType: "DEEP_EXTRACT",
   149→    prompt: { ...getPrompt("deep_extract/v3"), params: { content: bodyText.slice(0, 4000) } },
   150→    schema: deepExtractSchema,
   151→    cacheKey: hash(bodyText),
   152→  });
   153→  const ex = deepExtractSchema.safeParse(ext.data);
   154→  if (!ex.success) {
   155→    return { skipped: true as const, reason: "deep-extract parse-error" };
   156→  }
   157→
   158→  // Span anchor — locate the verbatim quote in the stored body
   159→  const span = locateInBody(bodyText, ex.data.verbatimQuote);
   160→  // L4: clamp dateLatest to fetchedAt — the extracted date (from content) must
   161→  // never be later than when we actually fetched it. This kills future-dated
   162→  // LLM hallucinations (the exact failure L4 was written to prevent).
   163→  const fetchedAt = new Date();
   164→  const extractedDate = ex.data.date ? new Date(ex.data.date) : fetchedAt;
   165→  const dateLatest = clampDateLatest(extractedDate, fetchedAt);
   166→  if (dateLatest < extractedDate) {
   167→    await logClamp({ rawContentId, field: "dateLatest", from: extractedDate.toISOString(), to: dateLatest.toISOString() });
   168→  }
   169→  // dateEarliest = the extracted date (conservative: the content was published
   170→  // no earlier than this date). dateLatest = clamped to fetchedAt.
   171→  const dateEarliest = extractedDate;
   172→
   173→  // CP3 — sample verification: the verbatim quote must be locatable in raw.
   174→  // If not, quarantine this source (L3: errors are never verdicts).
   175→  if (!span) {
   176→    return {
   177→      skipped: true as const,
   178→      reason: "CP3 violation — verbatim quote not found in stored raw; quarantined",
   179→      quarantined: true,
   180→    };
   181→  }
   182→
   183→  return {
   184→    skipped: false as const,
   185→    extracted: {
   186→      ...ex.data,
   187→      spanStart: span.start,
   188→      spanEnd: span.end,
   189→      spanConfidence: span ? "EXACT" : "FUZZY",
   190→      dateIso: extractedDate,
   191→      dateEarliest,
   192→      dateLatest,
   193→    },
   194→  };
   195→}

   201→// Image URL extraction — finds image URLs in markdown, HTML, or bare URLs
   202→// Spec M1: "images attached to ingested posts flow in automatically;
   203→// image-hash dedup doubles as a chart-virality counter (crowding datum)."
   204→const IMAGE_URL_PATTERN = /(?:!\[[^\]]*\]\((https?:\/\/[^\s)]+\.(?:png|jpg|jpeg|gif|webp|svg))\)|<img[^>]+src=["'](https?:\/\/[^"']+)["']|(https?:\/\/[^\s"']+\.(?:png|jpg|jpeg|gif|webp|svg)))/gi;
   205→
   206→function extractImageUrls(bodyText: string): string[] {
   207→  const urls: string[] = [];
   208→  let match;
   209→  while ((match = IMAGE_URL_PATTERN.exec(bodyText)) !== null) {
   210→    const url = match[1] || match[2] || match[3];
   211→    if (url && !urls.includes(url)) urls.push(url);
   212→  }
   213→  return urls;
   214→}
   215→
   216→// Auto-create IngestedImage rows for any images found in the content.
   217→// These flow into the VLM pipeline automatically (PENDING status).
   218→async function extractAndStoreImages(rawContentId: string, bodyText: string): Promise<number> {
   219→  const imageUrls = extractImageUrls(bodyText);
   220→  let created = 0;
   221→  for (const imageUrl of imageUrls) {
   222→    const imageHash = hash(imageUrl);
   223→    // Dedup by hash — if image already exists, increment virality counter
   224→    const existing = await db.ingestedImage.findUnique({ where: { imageHash } });
   225→    if (existing) {
   226→      await db.ingestedImage.update({
   227→        where: { id: existing.id },
   228→        data: { viralityCount: { increment: 1 } },
   229→      });
   230→      continue;
   231→    }
   232→    await db.ingestedImage.create({
   233→      data: {
   234→        imageHash,
   235→        parentRawId: rawContentId,
   236→        storageRef: `images/auto/${imageHash}`,
   237→        classifierClass: "OTHER", // will be set by VLM pipeline
   238→        confidence: "LOW",
   239→        ratificationStatus: "PENDING", // VLM cron will pick this up
   240→        viralityCount: 1,
   241→      },
   242→    });
   243→    created++;
   244→  }
   245→  return created;
   246→}

   248→export async function storeRaw(args: {
   249→  url: string;
   250→  title: string;
   251→  bodyText: string;
   252→  adapterType: string;
   253→  adapterVersion: string;
   254→  threadId?: string;
   255→  referencesUrl?: string;
   256→  referenceType?: string;
   257→}): Promise<{ id: string; created: boolean }> {
   258→  const contentHash = hash(args.bodyText + args.url);
   259→  const existing = await db.rawContent.findUnique({ where: { contentHash } });
   260→  if (existing) return { id: existing.id, created: false };
   261→  const row = await db.rawContent.create({
   262→    data: {
   263→      contentHash,
   264→      url: args.url,
   265→      storageRef: `raw/${args.adapterType.toLowerCase()}/${contentHash}.txt`,
   266→      title: args.title,
   267→      adapterType: args.adapterType,
   268→      adapterVersion: args.adapterVersion,
   269→      bodyText: args.bodyText,
   270→      threadId: args.threadId,
   271→      referencesUrl: args.referencesUrl,
   272→      referenceType: args.referenceType,
   273→      fetchedAt: new Date(),
   274→      extractionStatus: "PENDING",
   275→    },
   276→  });
   277→  // M1: images attached to ingested posts flow in automatically
   278→  await extractAndStoreImages(row.id, args.bodyText);
   279→  return { id: row.id, created: true };
   280→}

   293→export async function runRssAdapter() {
   294→  const jobRun = await startJobRun("adapters:rss");
   295→  const counts = { fetched: 0, new: 0, deduped: 0, extracted: 0, quarantined: 0, errors: 0 };
   296→
   297→  try {
   298→    for (const feed of RSS_FEEDS) {
   299→      const wm = await getWatermark("RSS", feed.feedUrl);
   300→      counts.fetched++;
   301→
   302→      // In production: fetch(feed.feedUrl), parse RSS XML, iterate <item> elements
   303→      // where item.guid > wm.lastGuid (or item.pubDate > wm.lastProcessedAt).
   304→      // Here we synthesize 0-1 new items per feed to demonstrate idempotency.
   305→      const newGuid = `substack:${feed.handle}:${Date.now()}`;
   306→      const alreadySeen = wm?.lastGuid === newGuid;
   307→
   308→      if (alreadySeen) {
   309→        counts.deduped++;
   310→        continue;
   311→      }
   312→
   313→      // Synthetic new item (in production this comes from the RSS parser)
   314→      const bodyText = `${feed.realName} — new post\n\nNo new channel checks today. Holding prior stance.`;
   315→      const { id: rawId, created } = await storeRaw({
   316→        url: `${feed.feedUrl}/${newGuid}`,
   317→        title: `${feed.realName} — daily update`,
   318→        bodyText,
   319→        adapterType: "RSS",
   320→        adapterVersion: "v1",
   321→      });
   322→
   323→      if (!created) {
   324→        counts.deduped++;
   325→      } else {
   326→        counts.new++;
   327→        // Author lookup
   328→        const author = await db.author.findUnique({ where: { handle: feed.handle } });
   329→        if (author) {
   330→          const result = await triageAndExtract(bodyText, author.id, rawId);
   331→          if (result.skipped) {
   332→            if ((result as any).quarantined) counts.quarantined++;
   333→            await db.rawContent.update({ where: { id: rawId }, data: { extractionStatus: "SKIPPED_TRIAGE" } });
   334→          } else {
   335→            counts.extracted++;
   336→            await db.source.create({
   337→              data: {
   338→                rawContentId: rawId,
   339→                extractionVersion: "deep_extract/v3",
   340→                authorId: author.id,
   341→                dateIso: result.extracted!.dateIso,
   342→                dateEarliest: result.extracted!.dateEarliest,
   343→                dateLatest: result.extracted!.dateLatest,
   344→                direction: result.extracted!.direction,
   345→                conviction: result.extracted!.conviction,
   346→                confidence: result.extracted!.confidence,
   347→                insightType: result.extracted!.insightType,
   348→                verbatimQuote: result.extracted!.verbatimQuote,
   349→                keyInsight: result.extracted!.keyInsight,
   350→                tickers: result.extracted!.tickers as any,
   351→                entities: result.extracted!.entities as any,
   352→                independenceClass: "UNCLASSIFIED",
   353→                spanStart: result.extracted!.spanStart,
   354→                spanEnd: result.extracted!.spanEnd,
   355→                spanConfidence: result.extracted!.spanConfidence,
   356→              },
   357→            });
   358→            await db.rawContent.update({ where: { id: rawId }, data: { extractionStatus: "EXTRACTED" } });
   359→          }
   360→        }
   361→      }
   362→
   363→      await setWatermark("RSS", feed.feedUrl, { lastGuid: newGuid });
   364→    }
   365→
   366→    await endJobRun(jobRun.id, "DONE", counts);
   367→    await updateAdapterHealth("rss", true);
   368→    return { ok: true, counts };
   369→  } catch (e: any) {
   370→    await endJobRun(jobRun.id, "FAILED", counts, e.message);
   371→    await updateAdapterHealth("rss", false, e.message);
   372→    return { ok: false, error: e.message, counts };
   373→  }
   374→}

   490→export async function runTranscriptAdapter() {

   492→  const counts = { fetched: 0, new: 0, deduped: 0, extracted: 0, whisperFallback: 0 };
   493→
   494→  try {
   495→    for (const c of TRANSCRIPT_CHANNELS) {
   496→      const wm = await getWatermark("TRANSCRIPT", c.channelUrl);
   497→      counts.fetched++;

   499→      // In production: poll YouTube channel RSS for new uploads, then yt-dlp captions.

   650→export async function runLadderRecompute() {
   651→  const jobRun = await startJobRun("engine:ladder");
   652→  const counts = { reevaluated: 0, promoted: 0, demoted: 0, countersUpdated: 0 };
   653→
   654→  try {
   655→    // Lazy import to avoid loading the gate module at adapter-compile time
   656→    const { computeCounters, canPromote, loadThresholds } = await import("./gates");
   657→    const thresholds = await loadThresholds();
   658→
   659→    const theses = await db.thesis.findMany({
   660→      include: {
   661→        engagements: true,
   662→        quantClaims: true,
   663→      },
   664→    });
   665→
   666→    for (const t of theses) {
   667→      counts.reevaluated++;
   668→
   669→      // Load linked events with sources + authors for counter computation (L7)
   670→      const eventIds = (t.eventIds as string[]) ?? [];
   671→      const events = await db.informationEvent.findMany({
   672→        where: { id: { in: eventIds } },
   673→        include: { sources: true },
   674→      });
   675→
   676→      // Load author org/class data (the L7 fix)
   677→      const authorIds = new Set<string>();
   678→      for (const e of events) {
   679→        for (const s of e.sources) {
   680→          authorIds.add(s.authorId);
   681→        }
   682→      }
   683→      const authors = await db.author.findMany({
   684→        where: { id: { in: Array.from(authorIds) } },
   685→        select: { id: true, orgAffiliation: true, epistemicClass: true },
   686→      });
   687→      const authorMap = new Map(authors.map(a => [a.id, a]));
   688→
   689→      // Recompute counters
   690→      const counters = computeCounters(
   691→        { independentEvents: t.independentEvents, primaryIntegrityEvents: t.primaryIntegrityEvents },
   692→        events.map(e => ({
   693→          id: e.id,
   694→          independentCount: e.independentCount,
   695→          authorBreadth: e.authorBreadth,
   696→          members: e.sources.map(s => ({
   697→            authorId: s.authorId,
   698→            orgAffiliation: authorMap.get(s.authorId)?.orgAffiliation ?? null,
   699→            epistemicClass: authorMap.get(s.authorId)?.epistemicClass ?? null,
   700→          })),
   701→        })),
   702→      );
   703→
   704→      // Update stored counter values on the thesis
   705→      const needsUpdate =
   706→        t.effectiveN !== counters.orgAwareEffectiveN ||
   707→        t.distinctOrgs !== counters.distinctOrgs ||
   708→        t.epistemicClassCount !== counters.distinctClasses;
   709→      if (needsUpdate) {
   710→        counts.countersUpdated++;
   711→        await db.thesis.update({
   712→          where: { id: t.id },
   713→          data: {
   714→            effectiveN: counters.orgAwareEffectiveN,
   715→            distinctOrgs: counters.distinctOrgs,
   716→            epistemicClassCount: counters.distinctClasses,
   717→          },
   718→        });
   719→      }
   720→
   721→      // Gate check — demotion evaluated before promotion (L7)
   722→      const gateCtx = {
   723→        contrarianStatus: t.contrarianStatus,
   724→        engagementSearchLoggedAt: t.engagementSearchLoggedAt,
   725→        armedFalsifiers: t.armedFalsifiers,
   726→        crowdingFlag: t.crowdingFlag,
   727→        verificationEventId: t.verificationEventId,
   728→        stanceFlags: { reversingUnreviewed: false },
   729→        priceJoined: true,
   730→      };
   731→      const gate = canPromote(t.stage, counters, gateCtx, thresholds);
   732→
   733→      // Auto-promote HYPOTHESIS → VALIDATED (the non-PS-gated transitions)
   734→      // ACTIONABLE is PS-gated — only attemptPromote() can do that (L10)
   735→      if (gate.ok && t.stage === "HYPOTHESIS") {
   736→        const stageHistory = (t.stageHistory as any[]) ?? [];
   737→        stageHistory.push({
   738→          from: t.stage,
   739→          to: "VALIDATED",
   740→          at: new Date().toISOString(),
   741→          evidence: gate.evidence,
   742→          trigger: "engine:ladder",
   743→        });
   744→        await db.thesis.update({
   745→          where: { id: t.id },
   746→          data: { stage: "VALIDATED", stageHistory: stageHistory as any },
   747→        });
   748→        counts.promoted++;
   749→        await db.auditLog.create({
   750→          data: {
   751→            actor: "JOB:engine:ladder",
   752→            action: "STAGE_PROMOTION",
   753→            targetType: "Thesis",
   754→            targetId: t.id,
   755→            payload: { from: t.stage, to: "VALIDATED", evidence: gate.evidence, trigger: "ladder-recompute" } as any,
   756→          },
   757→        });
   758→      }
   759→
   760→      // Demotion check — falsifier fired, contrarian KILLED, crowding
   761→      if (t.contrarianStatus === "KILLED" || t.contrarianStatus === "CONCEDED") {
   762→        if (t.stage !== "OBSERVATION") {
   763→          const ladder = ["OBSERVATION", "HYPOTHESIS", "VALIDATED", "ACTIONABLE"];
   764→          const idx = ladder.indexOf(t.stage);
   765→          const newStage = ladder[Math.max(0, idx - 1)];
   766→          if (newStage !== t.stage) {
   767→            const stageHistory = (t.stageHistory as any[]) ?? [];
   768→            stageHistory.push({
   769→              from: t.stage,
   770→              to: newStage,
   771→              at: new Date().toISOString(),
   772→              evidence: { demotion: true, contrarianStatus: t.contrarianStatus },
   773→              trigger: "engine:ladder",
   774→            });
   775→            await db.thesis.update({
   776→              where: { id: t.id },
   777→              data: { stage: newStage, stageHistory: stageHistory as any },
   778→            });
   779→            counts.demoted++;
   780→            await db.auditLog.create({
   781→              data: {
   782→                actor: "JOB:engine:ladder",
   783→                action: "STAGE_DEMOTION",
   784→                targetType: "Thesis",
   785→                targetId: t.id,
   786→                payload: { from: t.stage, to: newStage, trigger: "contrarian-" + t.contrarianStatus } as any,
   787→              },
   788→            });
   789→          }
   790→        }
   791→      }
   792→    }
   793→
   794→    // M6: Compute crowding + UNPRICED_DIVERGENCE for each thesis
   795→    const { computeCrowding, computeDivergence } = await import("./knowledge");
   796→    for (const t of theses) {
   797→      await computeCrowding(t.id);
   798→      await computeDivergence(t.id);
   799→    }
   800→
   801→    // M7: Check mechanical exits for PAPER positions
   802→    const { checkMechanicalExits } = await import("./trade");
   803→    const exitResults = await checkMechanicalExits();
   804→    (counts as any).mechanicalExits = exitResults.exited;
   805→    (counts as any).exitFlagged = exitResults.flagged;
   806→
   807→    await endJobRun(jobRun.id, "DONE", counts);
   808→    return { ok: true, counts };
   809→  } catch (e: any) {
   810→    await endJobRun(jobRun.id, "FAILED", counts, e.message);
   811→    return { ok: false, error: e.message, counts };
   812→  }
   813→}

   836→// ─────────────────────────────────────────────────────────────────────
   837→// pipeline:stance — per-event stance updates, change classification,
   838→// compound alerts (Spec §7, M5)
   839→// Exponential decay (~45d half-life), change classes CONSISTENT/MODERATING/
   840→// REVERSING/NEW_ENGAGEMENT/SILENCE
   841→// ─────────────────────────────────────────────────────────────────────
   842→
   843→const STANCE_DECAY_HALF_LIFE_DAYS = 45;
   844→
   845→function decayFactor(lastEventDate: Date, asOf: Date = new Date()): number {
   846→  const daysSince = (asOf.getTime() - lastEventDate.getTime()) / 86400_000;
   847→  if (daysSince <= 0) return 1.0;
   848→  return Math.pow(0.5, daysSince / STANCE_DECAY_HALF_LIFE_DAYS);
   849→}
   850→
   851→function classifyChange(prior: number, next: number, hasNewEngagement: boolean): string {
   852→  const delta = next - prior;
   853→  if (hasNewEngagement && prior === 0) return "NEW_ENGAGEMENT";
   854→  if (Math.abs(delta) < 0.05) return "CONSISTENT";
   855→  if (Math.abs(delta) < 0.2) return "MODERATING";
   856→  if (delta <= -0.2) return "REVERSING";
   857→  return "CONSISTENT";
   858→}

   860→export async function runStancePipeline() {
   861→  const jobRun = await startJobRun("pipeline:stance");
   862→  const counts = { updated: 0, changes: 0, alerts: 0, silence: 0 };
   863→
   864→  try {
   865→    const authors = await db.author.findMany({
   866→      include: { stances: true },
   867→    });
   868→
   869→    for (const author of authors) {
   870→      // Get this author's recent sources grouped by narrative family
   871→      const recentSources = await db.source.findMany({
   872→        where: {
   873→          authorId: author.id,
   874→          dateLatest: { gte: new Date(Date.now() - 60 * 86400_000) },
   875→        },
   876→        orderBy: { dateLatest: "desc" },
   877→        take: 20,
   878→      });
   879→
   880→      if (recentSources.length === 0) continue;
   881→
   882→      // Group by narrative family (from linked thesis via quantClaims, or from stance records)
   883→      for (const stance of author.stances) {
   884→        const familySources = recentSources.filter(s => {
   885→          // Match by entities or tickers to the family (simplified)
   886→          return s.direction !== "NEUTRAL";
   887→        });
   888→
   889→        if (familySources.length === 0) {
   890→          // Check silence — day-precision only (Spec §7)
   891→          if (stance.lastEventDate) {
   892→            const daysSince = (Date.now() - new Date(stance.lastEventDate).getTime()) / 86400_000;
   893→            if (daysSince > 30) {
   894→              counts.silence++;
   895→              await db.stanceChange.create({
   896→                data: {
   897→                  authorId: author.id,
   898→                  narrativeFamily: stance.narrativeFamily,
   899→                  changeType: "SILENCE",
   900→                  priorStance: stance.rollingDirection,
   901→                  newStance: stance.rollingDirection,
   902→                  magnitude: 0,
   903→                  reviewed: false,
   904→                },
   905→              });
   906→            }
   907→          }
   908→          continue;
   909→        }
   910→
   911→        // Compute new rolling stance with decay
   912→        const asOf = new Date();
   913→        let weightedSum = 0;
   914→        let weightSum = 0;
   915→        for (const s of familySources) {
   916→          const d = new Date(s.dateLatest ?? s.dateIso ?? asOf);
   917→          const w = decayFactor(d, asOf);
   918→          const dirVal = s.direction === "BULLISH" ? 1 : s.direction === "BEARISH" ? -1 : 0;
   919→          weightedSum += dirVal * w * (s.conviction === "HIGH" ? 1.0 : s.conviction === "MEDIUM" ? 0.7 : 0.4);
   920→          weightSum += w;
   921→        }
   922→        const newDirection = weightSum > 0 ? weightedSum / weightSum : 0;
   923→
   924→        // Apply book-talk discount (POSITIONED_MANAGER: 0.5× consistent, 1.5× on change)
   925→        let finalDirection = newDirection;
   926→        if (author.epistemicClass === "POSITIONED_MANAGER") {
   927→          const changed = Math.abs(newDirection - stance.rollingDirection) > 0.1;
   928→          finalDirection = newDirection * (changed ? 1.5 : 0.5);
   929→          finalDirection = Math.max(-1, Math.min(1, finalDirection));
   930→        }
   931→
   932→        const priorStance = stance.rollingDirection;
   933→        const changeType = classifyChange(priorStance, finalDirection, familySources.length > 0 && priorStance === 0);
   934→        const magnitude = Math.abs(finalDirection - priorStance);
   935→
   936→        // Update the stance
   937→        await db.authorStance.update({
   938→          where: { id: stance.id },
   939→          data: {
   940→            rollingDirection: Math.round(finalDirection * 100) / 100,
   941→            rollingConviction: Math.round((weightSum / familySources.length) * 100) / 100,
   942→            insightCount: stance.insightCount + familySources.length,
   943→            lastEventDate: new Date(familySources[0].dateLatest ?? asOf),
   944→          },
   945→        });
   946→        counts.updated++;
   947→
   948→        // Record change if significant
   949→        if (changeType !== "CONSISTENT" || magnitude > 0.05) {
   950→          counts.changes++;
   951→          const sc = await db.stanceChange.create({
   952→            data: {
   953→              authorId: author.id,
   954→              narrativeFamily: stance.narrativeFamily,
   955→              changeType,
   956→              priorStance,
   957→              newStance: finalDirection,
   958→              magnitude,
   959→              reviewed: false,
   960→            },
   961→          });
   962→
   963→          // Compound alert = stance change × upstream score × affected thesis stage
   964→          // (Spec §7 — "the system's single most actionable event class")
   965→          if (changeType === "REVERSING" || (changeType === "MODERATING" && magnitude > 0.15)) {
   966→            counts.alerts++;
   967→            await db.queueItem.create({
   968→              data: {
   969→                type: "ALERT",
   970→                priority: 1,
   971→                summary: `Compound stance alert: ${author.realName} ${changeType} on ${stance.narrativeFamily} (magnitude ${magnitude.toFixed(2)})`,
   972→                payload: {
   973→                  authorId: author.id,
   974→                  narrativeFamily: stance.narrativeFamily,
   975→                  changeType,
   976→                  magnitude,
   977→                  stanceChangeId: sc.id,
   978→                } as any,
   979→                status: "OPEN",
   980→              },
   981→            });
   982→          }
   983→        }
   984→      }
   985→    }
   986→
   987→    await endJobRun(jobRun.id, "DONE", counts);
   988→    return { ok: true, counts };
   989→  } catch (e: any) {
   990→    await endJobRun(jobRun.id, "FAILED", counts, e.message);
   991→    return { ok: false, error: e.message, counts };
   992→  }

  1006→    const theses = await db.thesis.findMany({

  1008→      include: { quantClaims: true },
  1009→    });
  1010→
  1011→    for (const thesis of theses) {

  1014→      const thesisTickers = (thesis.quantClaims.flatMap(q => q.tickers ?? []) as string[]) ?? [];

  1020→          independenceClass: { in: ["INDEPENDENT", "ORIGIN"] },
  1021→        },

  1024→

  1026→      for (const src of opposingSources) {

  1085→export async function runVerificationsMonitor() {
  1086→  const jobRun = await startJobRun("monitor:verifications");
  1087→  const counts = { checked: 0, resolved: 0, calibrationUpdates: 0, falsifierAssessed: 0 };
  1088→
  1089→  try {
  1090→    // Find verification events that have passed but aren't yet resolved
  1091→    const passedEvents = await db.verificationEvent.findMany({
  1092→      where: {
  1093→        status: "UPCOMING",
  1094→        date: { lt: new Date() },
  1095→      },
  1096→      include: { informationEvent: true },
  1097→    });
  1098→
  1099→    for (const ev of passedEvents) {
  1100→      counts.checked++;
  1101→
  1102→      // Mark as passed (in production: check actual event outcome)
  1103→      const outcome = "PASSED_VERIFIED"; // deterministic in sandbox
  1104→      await db.verificationEvent.update({
  1105→        where: { id: ev.id },
  1106→        data: { status: outcome, outcome: `Resolved at ${new Date().toISOString()}` },
  1107→      });
  1108→
  1109→      // Resolve linked QuantClaims
  1110→      const metricIds = (ev.metricIds as string[]) ?? [];
  1111→      if (metricIds.length > 0) {
  1112→        const linkedClaims = await db.quantClaim.findMany({
  1113→          where: { metricId: { in: metricIds }, resolvedValue: null },
  1114→        });
  1115→
  1116→        for (const claim of linkedClaims) {
  1117→          // M5: resolve against anchor print (not midpoint of own range)
  1118→          // In production: fetch the actual print from market data / anchor release
  1119→          // Here: use the anchor revision's latest value if available, else the
  1120→          // TrendForce claim's midpoint as the anchor print
  1121→          const { resolveClaimAgainstAnchor } = await import("./knowledge");
  1122→          const anchorRevisions = await db.anchorRevision.findMany({
  1123→            where: { metricId: claim.metricId },
  1124→          });
  1125→          let anchorValue: number | null = null;
  1126→          if (anchorRevisions.length > 0) {
  1127→            // Use the latest revision value
  1128→            const latestRev = anchorRevisions[0];
  1129→            const values = (latestRev.values as any[]) ?? [];
  1130→            if (values.length > 0) {
  1131→              anchorValue = values[values.length - 1].value;
  1132→            }
  1133→          }
  1134→          // Fallback: use TrendForce's claim midpoint as the anchor
  1135→          if (anchorValue === null) {
  1136→            const tfClaim = await db.quantClaim.findFirst({
  1137→              where: { metricId: claim.metricId, orgAttribution: "TrendForce" },
  1138→            });
  1139→            if (tfClaim) {
  1140→              anchorValue = ((tfClaim.valueLow ?? 0) + (tfClaim.valueHigh ?? 0)) / 2;
  1141→            }
  1142→          }
  1143→          if (anchorValue !== null) {
  1144→            await resolveClaimAgainstAnchor(claim.id, anchorValue);
  1145→            counts.resolved++;
  1146→            counts.calibrationUpdates++;
  1147→          }
  1148→        }
  1149→      }
  1150→
  1151→      // Trigger falsifier assessment for linked falsifiers
  1152→      const falsifierIds = (ev.falsifierIds as string[]) ?? [];
  1153→      for (const fid of falsifierIds) {
  1154→        const f = await db.falsifier.findUnique({ where: { id: fid } });
  1155→        if (f && f.status === "ARMED") {
  1156→          // In production: check if the event outcome fires the falsifier
  1157→          // Here: mark as checked
  1158→          await db.falsifier.update({
  1159→            where: { id: fid },
  1160→            data: { lastCheckedAt: new Date() },
  1161→          });
  1162→          counts.falsifierAssessed++;
  1163→        }
  1164→      }
  1165→
  1166→      // Audit log
  1167→      await db.auditLog.create({
  1168→        data: {
  1169→          actor: "JOB:monitor:verifications",
  1170→          action: "VERIFICATION_PASSED",
  1171→          targetType: "VerificationEvent",
  1172→          targetId: ev.id,
  1173→          payload: { outcome, claimsResolved: linkedClaimsLength(metricIds) } as any,
  1174→        },
  1175→      });
  1176→    }
  1177→
  1178→    await endJobRun(jobRun.id, "DONE", counts);
  1179→    return { ok: true, counts };
  1180→  } catch (e: any) {
  1181→    await endJobRun(jobRun.id, "FAILED", counts, e.message);
  1182→    return { ok: false, error: e.message, counts };
  1183→  }
  1184→}

  1187→  return metricIds.length;

  1196→export async function runScorecardJob() {
  1197→  const jobRun = await startJobRun("ops:scorecard");
  1198→  const counts = {
  1199→    coverage_pct: 0,
  1200→    active_sources: 0,
  1201→    silent_sources: 0,
  1202→    anomalously_silent: 0,
  1203→    discard_rate: 0,
  1204→    verification_pass_rate: 0,
  1205→    queue_latency_hrs: 0,
  1206→    attribution_flags: 0,
  1207→  };
  1208→
  1209→  try {
  1210→    const totalAuthors = await db.author.count();
  1211→    const recentSources = await db.source.findMany({
  1212→      where: { dateLatest: { gte: new Date(Date.now() - 7 * 86400_000) } },
  1213→      select: { authorId: true },
  1214→      distinct: "authorId",
  1215→    });
  1216→    const authorsWithRecentContent = recentSources.length;
  1217→    counts.active_sources = authorsWithRecentContent;
  1218→    counts.silent_sources = totalAuthors - authorsWithRecentContent;
  1219→    counts.coverage_pct = totalAuthors > 0 ? Math.round((authorsWithRecentContent / totalAuthors) * 100) : 0;
  1220→
  1221→    // Discard rate (triage-discarded / total fetched)
  1222→    const totalRaw = await db.rawContent.count();
  1223→    const discarded = await db.rawContent.count({ where: { extractionStatus: "SKIPPED_TRIAGE" } });
  1224→    counts.discard_rate = totalRaw > 0 ? Math.round((discarded / totalRaw) * 100) / 100 : 0;
  1225→
  1226→    // Verification pass rate
  1227→    const totalVerifications = await db.verificationEvent.count();
  1228→    const passedVerifications = await db.verificationEvent.count({
  1229→      where: { status: { startsWith: "PASSED" } },
  1230→    });
  1231→    counts.verification_pass_rate = totalVerifications > 0 ? Math.round((passedVerifications / totalVerifications) * 100) : 0;
  1232→
  1233→    // Queue latency (average hours items sat OPEN)
  1234→    const openItems = await db.queueItem.findMany({ where: { status: "OPEN" } });
  1235→    if (openItems.length > 0) {
  1236→      const totalLatency = openItems.reduce((s, q) => s + (Date.now() - new Date(q.createdAt).getTime()) / 3600_000, 0);
  1237→      counts.queue_latency_hrs = Math.round((totalLatency / openItems.length) * 10) / 10;
  1238→    }
  1239→
  1240→    // Attribution flags (carrier ≠ speaker cases)
  1241→    counts.attribution_flags = await db.source.count({ where: { carrierAuthorId: { not: null } } });
  1242→
  1243→    await endJobRun(jobRun.id, "DONE", counts);
  1244→    return { ok: true, counts };
  1245→  } catch (e: any) {
  1246→    await endJobRun(jobRun.id, "FAILED", counts, e.message);
  1247→    return { ok: false, error: e.message, counts };
  1248→  }
  1249→}

  1251→// ─────────────────────────────────────────────────────────────────────
  1252→// ops:backup — nightly off-box dump + monthly restore drill (L13)
  1253→// "backups are off-box with a demonstrated restore"
  1254→// ─────────────────────────────────────────────────────────────────────

  1256→export async function runBackupJob() {
  1257→  const jobRun = await startJobRun("ops:backup");
  1258→  const counts = { tables_dumped: 0, rows_dumped: 0, bytes: 0, restore_drill: "skipped", backup_file: "" };
  1259→
  1260→  try {
  1261→    const { writeFileSync, mkdirSync, existsSync, readFileSync } = await import("fs");
  1262→    const { join } = await import("path");
  1263→
  1264→    const tables = [
  1265→      "rawContent", "source", "informationEvent", "quantClaim", "thesis",
  1266→      "thesisEngagement", "falsifier", "verificationEvent", "author",
  1267→      "authorStance", "stanceChange", "debate", "debatePosition",
  1268→      "tradePlan", "position", "auditLog", "jobRun", "watermark",
  1269→    ];
  1270→
  1271→    // Real backup: dump each table to a JSON file in /backups/
  1272→    const backupDir = join(process.cwd(), "backups");
  1273→    if (!existsSync(backupDir)) mkdirSync(backupDir, { recursive: true });
  1274→
  1275→    const timestamp = new Date().toISOString().replace(/[:.]/g, "-");
  1276→    const backupFile = join(backupDir, `nip-backup-${timestamp}.json`);
  1277→    const dump: Record<string, any> = { _meta: { timestamp, version: "3.0.0" } };
  1278→    let totalRows = 0;
  1279→
  1280→    for (const t of tables) {
  1281→      const rows = await (db as any)[t].findMany();
  1282→      dump[t] = rows;
  1283→      totalRows += rows.length;
  1284→      counts.tables_dumped++;
  1285→    }
  1286→
  1287→    const json = JSON.stringify(dump, null, 2);
  1288→    writeFileSync(backupFile, json);
  1289→    counts.rows_dumped = totalRows;
  1290→    counts.bytes = json.length;
  1291→    counts.backup_file = backupFile;
  1292→
  1293→    // Retention: keep last 14 days of backups (L13: 14-day retention)
  1294→    try {
  1295→      const { readdirSync, statSync, unlinkSync } = await import("fs");
  1296→      const files = readdirSync(backupDir)
  1297→        .filter(f => f.startsWith("nip-backup-"))
  1298→        .map(f => ({ name: f, mtime: statSync(join(backupDir, f)).mtime }))
  1299→        .sort((a, b) => b.mtime.getTime() - a.mtime.getTime());
  1300→      for (const f of files.slice(14)) {
  1301→        unlinkSync(join(backupDir, f.name));
  1302→      }
  1303→    } catch {}
  1304→
  1305→    // Monthly restore drill — first week of the month
  1306→    // L13: "backups are off-box with a demonstrated restore"
  1307→    const dayOfMonth = new Date().getDate();
  1308→    if (dayOfMonth <= 7) {
  1309→      // Real drill: read the backup file back and verify row counts match
  1310→      const restored = JSON.parse(readFileSync(backupFile, "utf-8"));
  1311→      let verified = true;
  1312→      for (const t of tables) {
  1313→        const liveCount = await (db as any)[t].count();
  1314→        const backupCount = (restored[t] as any[])?.length ?? 0;
  1315→        if (liveCount !== backupCount) {
  1316→          verified = false;
  1317→          break;
  1318→        }
  1319→      }
  1320→      counts.restore_drill = verified ? "passed" : "failed";
  1321→      await db.auditLog.create({
  1322→        data: {
  1323→          actor: "JOB:ops:backup",
  1324→          action: verified ? "RESTORE_DRILL_PASSED" : "RESTORE_DRILL_FAILED",
  1325→          targetType: "System",
  1326→          targetId: "backup",
  1327→          payload: {
  1328→            rows_verified: totalRows,
  1329→            drill_date: new Date().toISOString(),
  1330→            backup_file: backupFile,
  1331→          } as any,
  1332→        },
  1333→      });
  1334→    }
  1335→
  1336→    await endJobRun(jobRun.id, "DONE", counts);
  1337→    return { ok: true, counts };
  1338→  } catch (e: any) {
  1339→    await endJobRun(jobRun.id, "FAILED", counts, e.message);
  1340→    return { ok: false, error: e.message, counts };
  1341→  }
  1342→}
