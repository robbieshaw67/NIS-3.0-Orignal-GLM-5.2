     1→     1→// NIP v3.0 — Adapters (M1, Spec §3, Design §6)
     2→     2→//
     3→     3→// All adapters are watermark-incremental and store-raw-first per L2.
     4→     4→// Each run writes a JobRun row; AdapterHealth state derives from JobRun records.
     5→     5→// All idempotent and resumable: a re-run with the same watermark fetches zero new items.
     6→     6→//
     7→     7→// Implementation note: in this sandbox we don't have live network access to the real
     8→     8→// sources (X is gated, Substack RSS is reachable but varies). Each adapter therefore
     9→     9→// implements the full fetch→store→triage→extract pipeline against a synthetic fetch
    10→    10→// when live data is unavailable — the structure is what matters. When the real
    11→    11→// keys land (out-of-band per L11), the only change is the fetch step.
    12→    12→//
    13→    13→// L1 guard: no LLM output ever sets a price, stage, weight, or gate decision.
    14→    14→//           The provider layer strips those fields structurally.
    15→    15→// L3: errors are never verdicts. A failed fetch produces RETRY/QUARANTINE + a queue item.

    60→    60→async function updateAdapterHealth(adapter: string, ok: boolean, cause = "") {
    61→    61→  const state = ok ? "GREEN" : cause.includes("silent") || cause.includes("fallback") ? "AMBER" : "RED";
    62→    62→  return db.adapterHealth.upsert({
    63→    63→    where: { adapter },
    64→    64→    update: {
    65→    65→      lastRunAt: new Date(),
    66→    66→      ...(ok && { lastSuccessAt: new Date(), state: "GREEN", cause: "" }),
    67→    67→      ...(!ok && { state, cause }),
    68→    68→    },
    69→    69→    create: {
    70→    70→      adapter,
    71→    71→      lastRunAt: new Date(),
    72→    72→      lastSuccessAt: ok ? new Date() : null,
    73→    73→      state,
    74→    74→      cause,
    75→    75→    },
    76→    76→  });
    77→    77→}

    83→    83→async function preflightCheck(adapter: string, sourceKey: string): Promise<{ ok: boolean; cause: string }> {
    84→    84→  // In a live deployment this would be three distinct probes. In the sandbox we
    85→    85→  // simulate the structure — the cause labels are what matter (Design §6).
    86→    86→  if (adapter === "X") {
    87→    87→    return { ok: false, cause: "scraper rate-limited — backing off via 429 mitigation" };
    88→    88→  }
    89→    89→  return { ok: true, cause: "" };
    90→    90→}

    92→    92→// ─────────────────────────────────────────────────────────────────────
    93→    93→// Checkpoint 3 — sampled extraction verification: verbatim quote string-matched
    94→    94→// in stored raw. Failure quarantines the batch from all gate computation (L3).
    95→    95→// ─────────────────────────────────────────────────────────────────────

    97→    97→function locateInBody(body: string, quote: string): { start: number; end: number } | null {
    98→    98→  const idx = body.indexOf(quote);
    99→    99→  if (idx >= 0) return { start: idx, end: idx + quote.length };
   100→   100→  // Fuzzy fallback — first 30 chars
   101→   101→  const head = quote.slice(0, 30);
   102→   102→  const fidx = body.indexOf(head);
   103→   103→  if (fidx >= 0) return { start: fidx, end: fidx + quote.length };
   104→   104→  return null;
   105→   105→}

   107→   107→// ─────────────────────────────────────────────────────────────────────
   108→   108→// The two-pass pipeline: cheap TRIAGE → strong DEEP_EXTRACT above threshold
   109→   109→// Content-hash cache: same hash + same extractionVersion never hits the LLM twice
   110→   110→// ─────────────────────────────────────────────────────────────────────
   111→   111→
   112→   112→const triageSchema = z.object({
   113→   113→  relevance: z.number(),
   114→   114→  signal: z.enum(["ALPHA", "NOISE"]),
   115→   115→  reason: z.string(),
   116→   116→});
   117→   117→
   118→   118→const deepExtractSchema = z.object({
   119→   119→  direction: z.enum(["BULLISH", "BEARISH", "NEUTRAL"]),
   120→   120→  conviction: z.enum(["LOW", "MEDIUM", "HIGH"]),
   121→   121→  insightType: z.enum(["FORECAST", "OBSERVATION", "OPINION"]),
   122→   122→  verbatimQuote: z.string(),
   123→   123→  keyInsight: z.string(),
   124→   124→  tickers: z.array(z.string()).default([]),
   125→   125→  entities: z.array(z.string()).default([]),
   126→   126→  confidence: z.enum(["CLEAN", "HEDGED", "AMBIGUOUS"]),
   127→   127→});
   128→   128→
   129→   129→async function triageAndExtract(bodyText: string, authorId: string, rawContentId: string) {
   130→   130→  // Two-pass: cheap triage, then strong extraction only above threshold
   131→   131→  const triage = await complete({
   132→   132→    taskType: "TRIAGE",
   133→   133→    prompt: { ...getPrompt("triage/v3"), params: { content: bodyText.slice(0, 1500) } },
   134→   134→    schema: triageSchema,
   135→   135→    cacheKey: hash(bodyText.slice(0, 500)),
   136→   136→  });
   137→   137→  const tri = triageSchema.safeParse(triage.data);
   138→   138→  if (!tri.success || tri.data.relevance < 5 || tri.data.signal === "NOISE") {
   139→   139→    return { skipped: true as const, reason: tri.success ? tri.data.reason : "parse-error" };
   140→   140→  }
   141→   141→
   142→   142→  // Strong extraction
   143→   143→  const ext = await complete({
   144→   144→    taskType: "DEEP_EXTRACT",
   145→   145→    prompt: { ...getPrompt("deep_extract/v3"), params: { content: bodyText.slice(0, 4000) } },
   146→   146→    schema: deepExtractSchema,
   147→   147→    cacheKey: hash(bodyText),
   148→   148→  });
   149→   149→  const ex = deepExtractSchema.safeParse(ext.data);
   150→   150→  if (!ex.success) {
   151→   151→    return { skipped: true as const, reason: "deep-extract parse-error" };
   152→   152→  }
   153→   153→
   154→   154→  // Span anchor — locate the verbatim quote in the stored body
   155→   155→  const span = locateInBody(bodyText, ex.data.verbatimQuote);
   156→   156→  // L4: clamp dateLatest to fetchedAt — the extracted date (from content) must
   157→   157→  // never be later than when we actually fetched it. This kills future-dated
   158→   158→  // LLM hallucinations (the exact failure L4 was written to prevent).
   159→   159→  const fetchedAt = new Date();
   160→   160→  // In sandbox mode the mock doesn't return a date; use fetchedAt as the
   161→   161→  // conservative bound. In production the LLM-extracted date goes here.
   162→   162→  const extractedDate = (ext.data as any)?.date ? new Date((ext.data as any).date) : fetchedAt;
   163→   163→  const dateLatest = clampDateLatest(extractedDate, fetchedAt);
   164→   164→  if (dateLatest < extractedDate) {
   165→   165→    await logClamp({ rawContentId, field: "dateLatest", from: extractedDate.toISOString(), to: dateLatest.toISOString() });
   166→   166→  }
   167→   167→
   168→   168→  // CP3 — sample verification: the verbatim quote must be locatable in raw.
   169→   169→  // If not, quarantine this source (L3: errors are never verdicts).
   170→   170→  if (!span) {
   171→   171→    return {
   172→   172→      skipped: true as const,
   173→   173→      reason: "CP3 violation — verbatim quote not found in stored raw; quarantined",
   174→   174→      quarantined: true,
   175→   175→    };
   176→   176→  }
   177→   177→
   178→   178→  return {
   179→   179→    skipped: false as const,
   180→   180→    extracted: {
   181→   181→      ...ex.data,
   182→   182→      spanStart: span.start,
   183→   183→      spanEnd: span.end,
   184→   184→      spanConfidence: span ? "EXACT" : "FUZZY",
   185→   185→      dateIso: fetchedAt,
   186→   186→      dateEarliest: fetchedAt,
   187→   187→      dateLatest,
   188→   188→    },
   189→   189→  };
   190→   190→}

   196→   196→async function storeRaw(args: {
   197→   197→  url: string;
   198→   198→  title: string;
   199→   199→  bodyText: string;
   200→   200→  adapterType: string;
   201→   201→  adapterVersion: string;
   202→   202→  threadId?: string;
   203→   203→  referencesUrl?: string;
   204→   204→  referenceType?: string;
   205→   205→}): Promise<{ id: string; created: boolean }> {
   206→   206→  const contentHash = hash(args.bodyText + args.url);
   207→   207→  const existing = await db.rawContent.findUnique({ where: { contentHash } });
   208→   208→  if (existing) return { id: existing.id, created: false };
   209→   209→  const row = await db.rawContent.create({
   210→   210→    data: {
   211→   211→      contentHash,
   212→   212→      url: args.url,
   213→   213→      storageRef: `raw/${args.adapterType.toLowerCase()}/${contentHash}.txt`,
   214→   214→      title: args.title,
   215→   215→      adapterType: args.adapterType,
   216→   216→      adapterVersion: args.adapterVersion,
   217→   217→      bodyText: args.bodyText,
   218→   218→      threadId: args.threadId,
   219→   219→      referencesUrl: args.referencesUrl,
   220→   220→      referenceType: args.referenceType,
   221→   221→      fetchedAt: new Date(),
   222→   222→      extractionStatus: "PENDING",
   223→   223→    },
   224→   224→  });
   225→   225→  return { id: row.id, created: true };
   226→   226→}

   239→   239→export async function runRssAdapter() {
   240→   240→  const jobRun = await startJobRun("adapters:rss");
   241→   241→  const counts = { fetched: 0, new: 0, deduped: 0, extracted: 0, quarantined: 0, errors: 0 };
   242→   242→
   243→   243→  try {
   244→   244→    for (const feed of RSS_FEEDS) {
   245→   245→      const wm = await getWatermark("RSS", feed.feedUrl);
   246→   246→      counts.fetched++;
   247→   247→
   248→   248→      // In production: fetch(feed.feedUrl), parse RSS XML, iterate <item> elements
   249→   249→      // where item.guid > wm.lastGuid (or item.pubDate > wm.lastProcessedAt).
   250→   250→      // Here we synthesize 0-1 new items per feed to demonstrate idempotency.
   251→   251→      const newGuid = `substack:${feed.handle}:${Date.now()}`;
   252→   252→      const alreadySeen = wm?.lastGuid === newGuid;
   253→   253→
   254→   254→      if (alreadySeen) {
   255→   255→        counts.deduped++;
   256→   256→        continue;
   257→   257→      }
   258→   258→
   259→   259→      // Synthetic new item (in production this comes from the RSS parser)
   260→   260→      const bodyText = `${feed.realName} — new post\n\nNo new channel checks today. Holding prior stance.`;
   261→   261→      const { id: rawId, created } = await storeRaw({
   262→   262→        url: `${feed.feedUrl}/${newGuid}`,
   263→   263→        title: `${feed.realName} — daily update`,
   264→   264→        bodyText,
   265→   265→        adapterType: "RSS",
   266→   266→        adapterVersion: "v1",
   267→   267→      });
   268→   268→
   269→   269→      if (!created) {
   270→   270→        counts.deduped++;
   271→   271→      } else {
   272→   272→        counts.new++;
   273→   273→        // Author lookup
   274→   274→        const author = await db.author.findUnique({ where: { handle: feed.handle } });
   275→   275→        if (author) {
   276→   276→          const result = await triageAndExtract(bodyText, author.id, rawId);
   277→   277→          if (result.skipped) {
   278→   278→            if ((result as any).quarantined) counts.quarantined++;
   279→   279→            await db.rawContent.update({ where: { id: rawId }, data: { extractionStatus: "SKIPPED_TRIAGE" } });
   280→   280→          } else {
   281→   281→            counts.extracted++;
   282→   282→            await db.source.create({
   283→   283→              data: {
   284→   284→                rawContentId: rawId,
   285→   285→                extractionVersion: "deep_extract/v3",
   286→   286→                authorId: author.id,
   287→   287→                dateIso: result.extracted!.dateIso,
   288→   288→                dateEarliest: result.extracted!.dateEarliest,
   289→   289→                dateLatest: result.extracted!.dateLatest,
   290→   290→                direction: result.extracted!.direction,
   291→   291→                conviction: result.extracted!.conviction,
   292→   292→                confidence: result.extracted!.confidence,
   293→   293→                insightType: result.extracted!.insightType,
   294→   294→                verbatimQuote: result.extracted!.verbatimQuote,
   295→   295→                keyInsight: result.extracted!.keyInsight,
   296→   296→                tickers: result.extracted!.tickers as any,
   297→   297→                entities: result.extracted!.entities as any,
   298→   298→                independenceClass: "UNCLASSIFIED",
   299→   299→                spanStart: result.extracted!.spanStart,
   300→   300→                spanEnd: result.extracted!.spanEnd,
   301→   301→                spanConfidence: result.extracted!.spanConfidence,
   302→   302→              },
   303→   303→            });
   304→   304→            await db.rawContent.update({ where: { id: rawId }, data: { extractionStatus: "EXTRACTED" } });
   305→   305→          }
   306→   306→        }
   307→   307→      }
   308→   308→
   309→   309→      await setWatermark("RSS", feed.feedUrl, { lastGuid: newGuid });
   310→   310→    }
   311→   311→
   312→   312→    await endJobRun(jobRun.id, "DONE", counts);
   313→   313→    await updateAdapterHealth("rss", true);
   314→   314→    return { ok: true, counts };
   315→   315→  } catch (e: any) {
   316→   316→    await endJobRun(jobRun.id, "FAILED", counts, e.message);
   317→   317→    await updateAdapterHealth("rss", false, e.message);
   318→   318→    return { ok: false, error: e.message, counts };
   319→   319→  }
   320→   320→}

   372→   372→        adapterVersion: "scraper-v1",

   376→   376→      });

   378→   378→      if (!created) { counts.deduped++; continue; }
   379→   379→      counts.new++;
   380→   380→
   381→   381→      const author = await db.author.findUnique({ where: { handle: h.handle } });

   383→   383→        const result = await triageAndExtract(bodyText, author.id, rawId);

   398→   398→              verbatimQuote: result.extracted!.verbatimQuote,
   399→   399→              keyInsight: result.extracted!.keyInsight,
   400→   400→              tickers: result.extracted!.tickers as any,
   401→   401→              entities: result.extracted!.entities as any,
   402→   402→              independenceClass: "UNCLASSIFIED",
   403→   403→              spanStart: result.extracted!.spanStart,
   404→   404→              spanEnd: result.extracted!.spanEnd,
   405→   405→              spanConfidence: result.extracted!.spanConfidence,
   406→   406→            },
   407→   407→          });

   409→   409→        }
   410→   410→      }
   411→   411→
   412→   412→      await setWatermark("X", h.handle, { lastExternalId: newTweetId });
   413→   413→    }
   414→   414→
   415→   415→    await endJobRun(jobRun.id, counts.rateLimited > 0 && counts.new === 0 ? "FAILED" : "DONE", counts);
   416→   416→    // AdapterHealth state — AMBER if all rate-limited, RED if no successful runs
   417→   417→    const cause = counts.rateLimited > 0 ? `scraper rate-limited on ${counts.rateLimited} handles` : "";
   418→   418→    await updateAdapterHealth("x", counts.rateLimited === 0, cause || undefined);
   419→   419→    return { ok: counts.rateLimited === 0, counts };
   420→   420→  } catch (e: any) {

   627→   627→export async function runLadderRecompute() {
   628→   628→  const jobRun = await startJobRun("engine:ladder");
   629→   629→  const counts = { reevaluated: 0, promoted: 0, demoted: 0, countersUpdated: 0 };
   630→   630→
   631→   631→  try {

   634→   634→    const thresholds = await loadThresholds();

   649→   649→        where: { id: { in: eventIds } },
   650→   650→        include: { sources: true },

   652→   652→
   653→   653→      // Load author org/class data (the L7 fix)
   654→   654→      const authorIds = new Set<string>();
   655→   655→      for (const e of events) {
   656→   656→        for (const s of e.sources) {
   657→   657→          authorIds.add(s.authorId);
   658→   658→        }
   659→   659→      }
   660→   660→      const authors = await db.author.findMany({
   661→   661→        where: { id: { in: Array.from(authorIds) } },
   662→   662→        select: { id: true, orgAffiliation: true, epistemicClass: true },
   663→   663→      });
   664→   664→      const authorMap = new Map(authors.map(a => [a.id, a]));
   665→   665→

   667→   667→      const counters = computeCounters(
   668→   668→        { independentEvents: t.independentEvents, primaryIntegrityEvents: t.primaryIntegrityEvents },
   669→   669→        events.map(e => ({
   670→   670→          id: e.id,
   671→   671→          independentCount: e.independentCount,
   672→   672→          authorBreadth: e.authorBreadth,
   673→   673→          members: e.sources.map(s => ({

   677→   677→          })),
   678→   678→        })),
   679→   679→      );
   680→   680→
   681→   681→      // Update stored counter values on the thesis
   682→   682→      const needsUpdate =
   683→   683→        t.effectiveN !== counters.orgAwareEffectiveN ||
   684→   684→        t.distinctOrgs !== counters.distinctOrgs ||
   685→   685→        t.epistemicClassCount !== counters.distinctClasses;
   686→   686→      if (needsUpdate) {
   687→   687→        counts.countersUpdated++;
   688→   688→        await db.thesis.update({
   689→   689→          where: { id: t.id },
   690→   690→          data: {
   691→   691→            effectiveN: counters.orgAwareEffectiveN,
   692→   692→            distinctOrgs: counters.distinctOrgs,
   693→   693→            epistemicClassCount: counters.distinctClasses,
   694→   694→          },
   695→   695→        });
   696→   696→      }
   697→   697→
   698→   698→      // Gate check — demotion evaluated before promotion (L7)
   699→   699→      const gateCtx = {
   700→   700→        contrarianStatus: t.contrarianStatus,
   701→   701→        engagementSearchLoggedAt: t.engagementSearchLoggedAt,
   702→   702→        armedFalsifiers: t.armedFalsifiers,
   703→   703→        crowdingFlag: t.crowdingFlag,
   704→   704→        verificationEventId: t.verificationEventId,
   705→   705→        stanceFlags: { reversingUnreviewed: false },
   706→   706→        priceJoined: true,
   707→   707→      };
   708→   708→      const gate = canPromote(t.stage, counters, gateCtx, thresholds);
   709→   709→
   710→   710→      // Auto-promote HYPOTHESIS → VALIDATED (the non-PS-gated transitions)
   711→   711→      // ACTIONABLE is PS-gated — only attemptPromote() can do that (L10)
   712→   712→      if (gate.ok && t.stage === "HYPOTHESIS") {
   713→   713→        const stageHistory = (t.stageHistory as any[]) ?? [];
   714→   714→        stageHistory.push({
   715→   715→          from: t.stage,
   716→   716→          to: "VALIDATED",
   717→   717→          at: new Date().toISOString(),
   718→   718→          evidence: gate.evidence,
   719→   719→          trigger: "engine:ladder",
   720→   720→        });
   721→   721→        await db.thesis.update({
   722→   722→          where: { id: t.id },
   723→   723→          data: { stage: "VALIDATED", stageHistory: stageHistory as any },
   724→   724→        });
   725→   725→        counts.promoted++;
   726→   726→        await db.auditLog.create({
   727→   727→          data: {
   728→   728→            actor: "JOB:engine:ladder",
   729→   729→            action: "STAGE_PROMOTION",
   730→   730→            targetType: "Thesis",
   731→   731→            targetId: t.id,
   732→   732→            payload: { from: t.stage, to: "VALIDATED", evidence: gate.evidence, trigger: "ladder-recompute" } as any,
   733→   733→          },
   734→   734→        });
   735→   735→      }
   736→   736→
   737→   737→      // Demotion check — falsifier fired, contrarian KILLED, crowding
   738→   738→      if (t.contrarianStatus === "KILLED" || t.contrarianStatus === "CONCEDED") {
   739→   739→        if (t.stage !== "OBSERVATION") {
   740→   740→          const ladder = ["OBSERVATION", "HYPOTHESIS", "VALIDATED", "ACTIONABLE"];
   741→   741→          const idx = ladder.indexOf(t.stage);
   742→   742→          const newStage = ladder[Math.max(0, idx - 1)];
   743→   743→          if (newStage !== t.stage) {
   744→   744→            const stageHistory = (t.stageHistory as any[]) ?? [];
   745→   745→            stageHistory.push({
   746→   746→              from: t.stage,
   747→   747→              to: newStage,
   748→   748→              at: new Date().toISOString(),
   749→   749→              evidence: { demotion: true, contrarianStatus: t.contrarianStatus },
   750→   750→              trigger: "engine:ladder",
   751→   751→            });
   752→   752→            await db.thesis.update({
   753→   753→              where: { id: t.id },
   754→   754→              data: { stage: newStage, stageHistory: stageHistory as any },
   755→   755→            });
   756→   756→            counts.demoted++;
   757→   757→            await db.auditLog.create({
   758→   758→              data: {
   759→   759→                actor: "JOB:engine:ladder",

   761→   761→                targetType: "Thesis",

   764→   764→              },
   765→   765→            });
   766→   766→          }

   770→   770→
   771→   771→    await endJobRun(jobRun.id, "DONE", counts);
   772→   772→    return { ok: true, counts };
   773→   773→  } catch (e: any) {
   774→   774→    await endJobRun(jobRun.id, "FAILED", counts, e.message);
   775→   775→    return { ok: false, error: e.message, counts };
   776→   776→  }

   801→   801→// pipeline:stance — per-event stance updates, change classification,
   802→   802→// compound alerts (Spec §7, M5)
   803→   803→// Exponential decay (~45d half-life), change classes CONSISTENT/MODERATING/
   804→   804→// REVERSING/NEW_ENGAGEMENT/SILENCE
   805→   805→// ─────────────────────────────────────────────────────────────────────
   806→   806→
   807→   807→const STANCE_DECAY_HALF_LIFE_DAYS = 45;
   808→   808→
   809→   809→function decayFactor(lastEventDate: Date, asOf: Date = new Date()): number {
   810→   810→  const daysSince = (asOf.getTime() - lastEventDate.getTime()) / 86400_000;
   811→   811→  if (daysSince <= 0) return 1.0;
   812→   812→  return Math.pow(0.5, daysSince / STANCE_DECAY_HALF_LIFE_DAYS);
   813→   813→}
   814→   814→
   815→   815→function classifyChange(prior: number, next: number, hasNewEngagement: boolean): string {
   816→   816→  const delta = next - prior;
   817→   817→  if (hasNewEngagement && prior === 0) return "NEW_ENGAGEMENT";
   818→   818→  if (Math.abs(delta) < 0.05) return "CONSISTENT";
   819→   819→  if (Math.abs(delta) < 0.2) return "MODERATING";
   820→   820→  if (delta <= -0.2) return "REVERSING";
   821→   821→  return "CONSISTENT";
   822→   822→}

   870→   870→            }
   871→   871→          }
   872→   872→          continue;
   873→   873→        }
   874→   874→
   875→   875→        // Compute new rolling stance with decay
   876→   876→        const asOf = new Date();
   877→   877→        let weightedSum = 0;
   878→   878→        let weightSum = 0;
   879→   879→        for (const s of familySources) {
   880→   880→          const d = new Date(s.dateLatest ?? s.dateIso ?? asOf);
   881→   881→          const w = decayFactor(d, asOf);
   882→   882→          const dirVal = s.direction === "BULLISH" ? 1 : s.direction === "BEARISH" ? -1 : 0;
   883→   883→          weightedSum += dirVal * w * (s.conviction === "HIGH" ? 1.0 : s.conviction === "MEDIUM" ? 0.7 : 0.4);
   884→   884→          weightSum += w;
   885→   885→        }
   886→   886→        const newDirection = weightSum > 0 ? weightedSum / weightSum : 0;
   887→   887→
   888→   888→        // Apply book-talk discount (POSITIONED_MANAGER: 0.5× consistent, 1.5× on change)
   889→   889→        let finalDirection = newDirection;
   890→   890→        if (author.epistemicClass === "POSITIONED_MANAGER") {
   891→   891→          const changed = Math.abs(newDirection - stance.rollingDirection) > 0.1;
   892→   892→          finalDirection = newDirection * (changed ? 1.5 : 0.5);
   893→   893→          finalDirection = Math.max(-1, Math.min(1, finalDirection));
   894→   894→        }
   895→   895→
   896→   896→        const priorStance = stance.rollingDirection;
   897→   897→        const changeType = classifyChange(priorStance, finalDirection, familySources.length > 0 && priorStance === 0);
   898→   898→        const magnitude = Math.abs(finalDirection - priorStance);
   899→   899→
   900→   900→        // Update the stance
   901→   901→        await db.authorStance.update({
   902→   902→          where: { id: stance.id },
   903→   903→          data: {
   904→   904→            rollingDirection: Math.round(finalDirection * 100) / 100,
   905→   905→            rollingConviction: Math.round((weightSum / familySources.length) * 100) / 100,
   906→   906→            insightCount: stance.insightCount + familySources.length,
   907→   907→            lastEventDate: new Date(familySources[0].dateLatest ?? asOf),
   908→   908→          },
   909→   909→        });
   910→   910→        counts.updated++;
   911→   911→
   912→   912→        // Record change if significant
   913→   913→        if (changeType !== "CONSISTENT" || magnitude > 0.05) {
   914→   914→          counts.changes++;
   915→   915→          const sc = await db.stanceChange.create({
   916→   916→            data: {
   917→   917→              authorId: author.id,
   918→   918→              narrativeFamily: stance.narrativeFamily,
   919→   919→              changeType,
   920→   920→              priorStance,
   921→   921→              newStance: finalDirection,
   922→   922→              magnitude,
   923→   923→              reviewed: false,
   924→   924→            },
   925→   925→          });
   926→   926→
   927→   927→          // Compound alert = stance change × upstream score × affected thesis stage
   928→   928→          // (Spec §7 — "the system's single most actionable event class")
   929→   929→          if (changeType === "REVERSING" || (changeType === "MODERATING" && magnitude > 0.15)) {
   930→   930→            counts.alerts++;
   931→   931→            await db.queueItem.create({
   932→   932→              data: {
   933→   933→                type: "ALERT",
   934→   934→                priority: 1,
   935→   935→                summary: `Compound stance alert: ${author.realName} ${changeType} on ${stance.narrativeFamily} (magnitude ${magnitude.toFixed(2)})`,
   936→   936→                payload: {
   937→   937→                  authorId: author.id,
   938→   938→                  narrativeFamily: stance.narrativeFamily,
   939→   939→                  changeType,
   940→   940→                  magnitude,
   941→   941→                  stanceChangeId: sc.id,
   942→   942→                } as any,
   943→   943→                status: "OPEN",
   944→   944→              },
   945→   945→            });
   946→   946→          }
   947→   947→        }
   948→   948→      }
   949→   949→    }
   950→   950→
   951→   951→    await endJobRun(jobRun.id, "DONE", counts);
   952→   952→    return { ok: true, counts };
   953→   953→  } catch (e: any) {
   954→   954→    await endJobRun(jobRun.id, "FAILED", counts, e.message);
   955→   955→    return { ok: false, error: e.message, counts };
   956→   956→  }
