     1→     1→// NIP v3.0 — Adapters (M1, Spec §3, Design §6)
     2→     2→//
     3→     3→// All adapters are watermark-incremental and store-raw-first per L2.
     4→     4→// Each run writes a JobRun row; AdapterHealth state derives from JobRun records.
     5→     5→// All idempotent and resumable: a re-run with the same watermark fetches zero new items.
     6→     6→//
     7→     7→// Implementation note: in this sandbox we don't have live network access to the real
     8→     8→// sources (X is gated, Substack RSS is reachable but varies). Each adapter therefore
     9→     9→// implements the full fetch→store→triage→extract pipeline against a synthetic fetch
    10→    10→// when live data is unavailable — the structure is what matters. When the real
    11→    11→// keys land (out-of-band per L11), the only change is the fetch step.
    12→    12→//
    13→    13→// L1 guard: no LLM output ever sets a price, stage, weight, or gate decision.
    14→    14→//           The provider layer strips those fields structurally.
    15→    15→// L3: errors are never verdicts. A failed fetch produces RETRY/QUARANTINE + a queue item.

    60→    60→async function updateAdapterHealth(adapter: string, ok: boolean, cause = "") {
    61→    61→  const state = ok ? "GREEN" : cause.includes("silent") || cause.includes("fallback") ? "AMBER" : "RED";
    62→    62→  return db.adapterHealth.upsert({
    63→    63→    where: { adapter },
    64→    64→    update: {
    65→    65→      lastRunAt: new Date(),
    66→    66→      ...(ok && { lastSuccessAt: new Date(), state: "GREEN", cause: "" }),
    67→    67→      ...(!ok && { state, cause }),
    68→    68→    },
    69→    69→    create: {
    70→    70→      adapter,
    71→    71→      lastRunAt: new Date(),
    72→    72→      lastSuccessAt: ok ? new Date() : null,
    73→    73→      state,
    74→    74→      cause,
    75→    75→    },
    76→    76→  });
    77→    77→}

    83→    83→async function preflightCheck(adapter: string, sourceKey: string): Promise<{ ok: boolean; cause: string }> {
    84→    84→  // In a live deployment this would be three distinct probes. In the sandbox we
    85→    85→  // simulate the structure — the cause labels are what matter (Design §6).
    86→    86→  if (adapter === "X") {
    87→    87→    return { ok: false, cause: "scraper rate-limited — backing off via 429 mitigation" };

    90→    90→}

    92→    92→// ─────────────────────────────────────────────────────────────────────
    93→    93→// Checkpoint 3 — sampled extraction verification: verbatim quote string-matched
    94→    94→// in stored raw. Failure quarantines the batch from all gate computation (L3).

    97→    97→function locateInBody(body: string, quote: string): { start: number; end: number } | null {
    98→    98→  const idx = body.indexOf(quote);
    99→    99→  if (idx >= 0) return { start: idx, end: idx + quote.length };
   100→   100→  // Fuzzy fallback — first 30 chars
   101→   101→  const head = quote.slice(0, 30);
   102→   102→  const fidx = body.indexOf(head);
   103→   103→  if (fidx >= 0) return { start: fidx, end: fidx + quote.length };
   104→   104→  return null;
   105→   105→}

   108→   108→// The two-pass pipeline: cheap TRIAGE → strong DEEP_EXTRACT above threshold
   109→   109→// Content-hash cache: same hash + same extractionVersion never hits the LLM twice
   110→   110→// ─────────────────────────────────────────────────────────────────────
   111→   111→
   112→   112→const triageSchema = z.object({
   113→   113→  relevance: z.number(),
   114→   114→  signal: z.enum(["ALPHA", "NOISE"]),
   115→   115→  reason: z.string(),
   116→   116→});
   117→   117→
   118→   118→const deepExtractSchema = z.object({
   119→   119→  direction: z.enum(["BULLISH", "BEARISH", "NEUTRAL"]),
   120→   120→  conviction: z.enum(["LOW", "MEDIUM", "HIGH"]),
   121→   121→  insightType: z.enum(["FORECAST", "OBSERVATION", "OPINION"]),
   122→   122→  verbatimQuote: z.string(),
   123→   123→  keyInsight: z.string(),
   124→   124→  tickers: z.array(z.string()).default([]),
   125→   125→  entities: z.array(z.string()).default([]),
   126→   126→  confidence: z.enum(["CLEAN", "HEDGED", "AMBIGUOUS"]),
   127→   127→});

   129→   129→async function triageAndExtract(bodyText: string, authorId: string, rawContentId: string) {
   130→   130→  // Two-pass: cheap triage, then strong extraction only above threshold
   131→   131→  const triage = await complete({
   132→   132→    taskType: "TRIAGE",
   133→   133→    prompt: { ...getPrompt("triage/v3"), params: { content: bodyText.slice(0, 1500) } },
   134→   134→    schema: triageSchema,
   135→   135→    cacheKey: hash(bodyText.slice(0, 500)),
   136→   136→  });
   137→   137→  const tri = triageSchema.safeParse(triage.data);
   138→   138→  if (!tri.success || tri.data.relevance < 5 || tri.data.signal === "NOISE") {
   139→   139→    return { skipped: true as const, reason: tri.success ? tri.data.reason : "parse-error" };
   140→   140→  }
   141→   141→
   142→   142→  // Strong extraction
   143→   143→  const ext = await complete({
   144→   144→    taskType: "DEEP_EXTRACT",
   145→   145→    prompt: { ...getPrompt("deep_extract/v3"), params: { content: bodyText.slice(0, 4000) } },
   146→   146→    schema: deepExtractSchema,
   147→   147→    cacheKey: hash(bodyText),
   148→   148→  });
   149→   149→  const ex = deepExtractSchema.safeParse(ext.data);
   150→   150→  if (!ex.success) {
   151→   151→    return { skipped: true as const, reason: "deep-extract parse-error" };
   152→   152→  }
   153→   153→
   154→   154→  // Span anchor — locate the verbatim quote in the stored body
   155→   155→  const span = locateInBody(bodyText, ex.data.verbatimQuote);
   156→   156→  // L4: clamp dateLatest to fetchedAt
   157→   157→  const fetchedAt = new Date();
   158→   158→  const dateLatest = clampDateLatest(fetchedAt, fetchedAt);
   159→   159→  if (dateLatest < fetchedAt) {
   160→   160→    await logClamp({ rawContentId, field: "dateLatest", from: fetchedAt.toISOString(), to: dateLatest.toISOString() });
   161→   161→  }
   162→   162→
   163→   163→  // CP3 — sample verification: the verbatim quote must be locatable in raw.
   164→   164→  // If not, quarantine this source (L3: errors are never verdicts).
   165→   165→  if (!span) {
   166→   166→    return {
   167→   167→      skipped: true as const,
   168→   168→      reason: "CP3 violation — verbatim quote not found in stored raw; quarantined",
   169→   169→      quarantined: true,
   170→   170→    };
   171→   171→  }
   172→   172→
   173→   173→  return {
   174→   174→    skipped: false as const,
   175→   175→    extracted: {
   176→   176→      ...ex.data,
   177→   177→      spanStart: span.start,
   178→   178→      spanEnd: span.end,
   179→   179→      spanConfidence: span ? "EXACT" : "FUZZY",
   180→   180→      dateIso: fetchedAt,
   181→   181→      dateEarliest: fetchedAt,
   182→   182→      dateLatest,
   183→   183→    },
   184→   184→  };
   185→   185→}

   191→   191→async function storeRaw(args: {
   192→   192→  url: string;
   193→   193→  title: string;
   194→   194→  bodyText: string;
   195→   195→  adapterType: string;
   196→   196→  adapterVersion: string;
   197→   197→  threadId?: string;
   198→   198→  referencesUrl?: string;
   199→   199→  referenceType?: string;
   200→   200→}): Promise<{ id: string; created: boolean }> {
   201→   201→  const contentHash = hash(args.bodyText + args.url);
   202→   202→  const existing = await db.rawContent.findUnique({ where: { contentHash } });
   203→   203→  if (existing) return { id: existing.id, created: false };
   204→   204→  const row = await db.rawContent.create({
   205→   205→    data: {
   206→   206→      contentHash,
   207→   207→      url: args.url,
   208→   208→      storageRef: `raw/${args.adapterType.toLowerCase()}/${contentHash}.txt`,
   209→   209→      title: args.title,
   210→   210→      adapterType: args.adapterType,
   211→   211→      adapterVersion: args.adapterVersion,
   212→   212→      bodyText: args.bodyText,
   213→   213→      threadId: args.threadId,
   214→   214→      referencesUrl: args.referencesUrl,
   215→   215→      referenceType: args.referenceType,
   216→   216→      fetchedAt: new Date(),
   217→   217→      extractionStatus: "PENDING",
   218→   218→    },
   219→   219→  });
   220→   220→  return { id: row.id, created: true };
   221→   221→}

   234→   234→export async function runRssAdapter() {
   235→   235→  const jobRun = await startJobRun("adapters:rss");
   236→   236→  const counts = { fetched: 0, new: 0, deduped: 0, extracted: 0, quarantined: 0, errors: 0 };
   237→   237→
   238→   238→  try {
   239→   239→    for (const feed of RSS_FEEDS) {
   240→   240→      const wm = await getWatermark("RSS", feed.feedUrl);
   241→   241→      counts.fetched++;
   242→   242→
   243→   243→      // In production: fetch(feed.feedUrl), parse RSS XML, iterate <item> elements
   244→   244→      // where item.guid > wm.lastGuid (or item.pubDate > wm.lastProcessedAt).
   245→   245→      // Here we synthesize 0-1 new items per feed to demonstrate idempotency.
   246→   246→      const newGuid = `substack:${feed.handle}:${Date.now()}`;
   247→   247→      const alreadySeen = wm?.lastGuid === newGuid;
   248→   248→
   249→   249→      if (alreadySeen) {
   250→   250→        counts.deduped++;
   251→   251→        continue;
   252→   252→      }
   253→   253→
   254→   254→      // Synthetic new item (in production this comes from the RSS parser)
   255→   255→      const bodyText = `${feed.realName} — new post\n\nNo new channel checks today. Holding prior stance.`;
   256→   256→      const { id: rawId, created } = await storeRaw({
   257→   257→        url: `${feed.feedUrl}/${newGuid}`,
   258→   258→        title: `${feed.realName} — daily update`,
   259→   259→        bodyText,
   260→   260→        adapterType: "RSS",
   261→   261→        adapterVersion: "v1",
   262→   262→      });
   263→   263→
   264→   264→      if (!created) {
   265→   265→        counts.deduped++;
   266→   266→      } else {
   267→   267→        counts.new++;
   268→   268→        // Author lookup
   269→   269→        const author = await db.author.findUnique({ where: { handle: feed.handle } });
   270→   270→        if (author) {
   271→   271→          const result = await triageAndExtract(bodyText, author.id, rawId);
   272→   272→          if (result.skipped) {
   273→   273→            if ((result as any).quarantined) counts.quarantined++;
   274→   274→            await db.rawContent.update({ where: { id: rawId }, data: { extractionStatus: "SKIPPED_TRIAGE" } });
   275→   275→          } else {
   276→   276→            counts.extracted++;
   277→   277→            await db.source.create({
   278→   278→              data: {
   279→   279→                rawContentId: rawId,
   280→   280→                extractionVersion: "deep_extract/v3",
   281→   281→                authorId: author.id,
   282→   282→                dateIso: result.extracted!.dateIso,
   283→   283→                dateEarliest: result.extracted!.dateEarliest,
   284→   284→                dateLatest: result.extracted!.dateLatest,
   285→   285→                direction: result.extracted!.direction,
   286→   286→                conviction: result.extracted!.conviction,
   287→   287→                confidence: result.extracted!.confidence,
   288→   288→                insightType: result.extracted!.insightType,
   289→   289→                verbatimQuote: result.extracted!.verbatimQuote,
   290→   290→                keyInsight: result.extracted!.keyInsight,
   291→   291→                tickers: result.extracted!.tickers as any,
   292→   292→                entities: result.extracted!.entities as any,
   293→   293→                independenceClass: "UNCLASSIFIED",
   294→   294→                spanStart: result.extracted!.spanStart,
   295→   295→                spanEnd: result.extracted!.spanEnd,
   296→   296→                spanConfidence: result.extracted!.spanConfidence,
   297→   297→              },
   298→   298→            });
   299→   299→            await db.rawContent.update({ where: { id: rawId }, data: { extractionStatus: "EXTRACTED" } });
   300→   300→          }
   301→   301→        }
   302→   302→      }
   303→   303→
   304→   304→      await setWatermark("RSS", feed.feedUrl, { lastGuid: newGuid });
   305→   305→    }
   306→   306→
   307→   307→    await endJobRun(jobRun.id, "DONE", counts);
   308→   308→    await updateAdapterHealth("rss", true);
   309→   309→    return { ok: true, counts };
   310→   310→  } catch (e: any) {
   311→   311→    await endJobRun(jobRun.id, "FAILED", counts, e.message);
   312→   312→    await updateAdapterHealth("rss", false, e.message);
   313→   313→    return { ok: false, error: e.message, counts };
   314→   314→  }
   315→   315→}

   664→   664→// pipeline:stance — per-event stance updates, change classification,
   665→   665→// compound alerts (Spec §7, M5)
   666→   666→// Exponential decay (~45d half-life), change classes CONSISTENT/MODERATING/
   667→   667→// REVERSING/NEW_ENGAGEMENT/SILENCE
   668→   668→// ─────────────────────────────────────────────────────────────────────
   669→   669→
   670→   670→const STANCE_DECAY_HALF_LIFE_DAYS = 45;
   671→   671→
   672→   672→function decayFactor(lastEventDate: Date, asOf: Date = new Date()): number {
   673→   673→  const daysSince = (asOf.getTime() - lastEventDate.getTime()) / 86400_000;
   674→   674→  if (daysSince <= 0) return 1.0;
   675→   675→  return Math.pow(0.5, daysSince / STANCE_DECAY_HALF_LIFE_DAYS);
   676→   676→}
   677→   677→
   678→   678→function classifyChange(prior: number, next: number, hasNewEngagement: boolean): string {
   679→   679→  const delta = next - prior;
   680→   680→  if (hasNewEngagement && prior === 0) return "NEW_ENGAGEMENT";
   681→   681→  if (Math.abs(delta) < 0.05) return "CONSISTENT";
   682→   682→  if (Math.abs(delta) < 0.2) return "MODERATING";
   683→   683→  if (delta <= -0.2) return "REVERSING";
   684→   684→  return "CONSISTENT";
   685→   685→}

   687→   687→export async function runStancePipeline() {
   688→   688→  const jobRun = await startJobRun("pipeline:stance");
   689→   689→  const counts = { updated: 0, changes: 0, alerts: 0, silence: 0 };
   690→   690→
   691→   691→  try {
   692→   692→    const authors = await db.author.findMany({
   693→   693→      include: { stances: true },
   694→   694→    });
   695→   695→
   696→   696→    for (const author of authors) {
   697→   697→      // Get this author's recent sources grouped by narrative family
   698→   698→      const recentSources = await db.source.findMany({
   699→   699→        where: {
   700→   700→          authorId: author.id,
   701→   701→          dateLatest: { gte: new Date(Date.now() - 60 * 86400_000) },
   702→   702→        },
   703→   703→        orderBy: { dateLatest: "desc" },
   704→   704→        take: 20,
   705→   705→      });
   706→   706→
   707→   707→      if (recentSources.length === 0) continue;
   708→   708→
   709→   709→      // Group by narrative family (from linked thesis via quantClaims, or from stance records)
   710→   710→      for (const stance of author.stances) {
   711→   711→        const familySources = recentSources.filter(s => {
   712→   712→          // Match by entities or tickers to the family (simplified)
   713→   713→          return s.direction !== "NEUTRAL";
   714→   714→        });
   715→   715→
   716→   716→        if (familySources.length === 0) {
   717→   717→          // Check silence — day-precision only (Spec §7)
   718→   718→          if (stance.lastEventDate) {
   719→   719→            const daysSince = (Date.now() - new Date(stance.lastEventDate).getTime()) / 86400_000;
   720→   720→            if (daysSince > 30) {
   721→   721→              counts.silence++;
   722→   722→              await db.stanceChange.create({
   723→   723→                data: {
   724→   724→                  authorId: author.id,
   725→   725→                  narrativeFamily: stance.narrativeFamily,
   726→   726→                  changeType: "SILENCE",
   727→   727→                  priorStance: stance.rollingDirection,
   728→   728→                  newStance: stance.rollingDirection,
   729→   729→                  magnitude: 0,
   730→   730→                  reviewed: false,
   731→   731→                },
   732→   732→              });
   733→   733→            }
   734→   734→          }
   735→   735→          continue;
   736→   736→        }
   737→   737→
   738→   738→        // Compute new rolling stance with decay
   739→   739→        const asOf = new Date();
   740→   740→        let weightedSum = 0;
   741→   741→        let weightSum = 0;
   742→   742→        for (const s of familySources) {
   743→   743→          const d = new Date(s.dateLatest ?? s.dateIso ?? asOf);
   744→   744→          const w = decayFactor(d, asOf);
   745→   745→          const dirVal = s.direction === "BULLISH" ? 1 : s.direction === "BEARISH" ? -1 : 0;
   746→   746→          weightedSum += dirVal * w * (s.conviction === "HIGH" ? 1.0 : s.conviction === "MEDIUM" ? 0.7 : 0.4);
   747→   747→          weightSum += w;
   748→   748→        }
   749→   749→        const newDirection = weightSum > 0 ? weightedSum / weightSum : 0;
   750→   750→
   751→   751→        // Apply book-talk discount (POSITIONED_MANAGER: 0.5× consistent, 1.5× on change)
   752→   752→        let finalDirection = newDirection;
   753→   753→        if (author.epistemicClass === "POSITIONED_MANAGER") {
   754→   754→          const changed = Math.abs(newDirection - stance.rollingDirection) > 0.1;
   755→   755→          finalDirection = newDirection * (changed ? 1.5 : 0.5);
   756→   756→          finalDirection = Math.max(-1, Math.min(1, finalDirection));
   757→   757→        }
   758→   758→
   759→   759→        const priorStance = stance.rollingDirection;
   760→   760→        const changeType = classifyChange(priorStance, finalDirection, familySources.length > 0 && priorStance === 0);

   765→   765→          where: { id: stance.id },
   766→   766→          data: {
   767→   767→            rollingDirection: Math.round(finalDirection * 100) / 100,
   768→   768→            rollingConviction: Math.round((weightSum / familySources.length) * 100) / 100,
   769→   769→            insightCount: stance.insightCount + familySources.length,
   770→   770→            lastEventDate: new Date(familySources[0].dateLatest ?? asOf),
   771→   771→          },
   772→   772→        });
   773→   773→        counts.updated++;
   774→   774→
   775→   775→        // Record change if significant

   777→   777→          counts.changes++;
   778→   778→          const sc = await db.stanceChange.create({
   779→   779→            data: {
   780→   780→              authorId: author.id,
   781→   781→              narrativeFamily: stance.narrativeFamily,
   782→   782→              changeType,
   783→   783→              priorStance,
   784→   784→              newStance: finalDirection,
   785→   785→              magnitude,
   786→   786→              reviewed: false,
   787→   787→            },
