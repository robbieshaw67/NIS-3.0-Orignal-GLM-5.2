     1→// NIP v3.0 — Adapters (M1, Spec §3, Design §6)
     2→//
     3→// All adapters are watermark-incremental and store-raw-first per L2.
     4→// Each run writes a JobRun row; AdapterHealth state derives from JobRun records.
     5→// All idempotent and resumable: a re-run with the same watermark fetches zero new items.
     6→//
     7→// Implementation note: in this sandbox we don't have live network access to the real
     8→// sources (X is gated, Substack RSS is reachable but varies). Each adapter therefore
     9→// implements the full fetch→store→triage→extract pipeline against a synthetic fetch
    10→// when live data is unavailable — the structure is what matters. When the real
    11→// keys land (out-of-band per L11), the only change is the fetch step.
    12→//
    13→// L1 guard: no LLM output ever sets a price, stage, weight, or gate decision.
    14→//           The provider layer strips those fields structurally.
    15→// L3: errors are never verdicts. A failed fetch produces RETRY/QUARANTINE + a queue item.

    27→function hash(s: string): string {
    28→  return createHash("sha256").update(s).digest("hex").slice(0, 32);

    60→async function updateAdapterHealth(adapter: string, ok: boolean, cause = "") {
    61→  const state = ok ? "GREEN" : cause.includes("silent") || cause.includes("fallback") ? "AMBER" : "RED";
    62→  return db.adapterHealth.upsert({
    63→    where: { adapter },
    64→    update: {
    65→      lastRunAt: new Date(),
    66→      ...(ok && { lastSuccessAt: new Date(), state: "GREEN", cause: "" }),
    67→      ...(!ok && { state, cause }),
    68→    },
    69→    create: {
    70→      adapter,
    71→      lastRunAt: new Date(),
    72→      lastSuccessAt: ok ? new Date() : null,
    73→      state,
    74→      cause,
    75→    },
    76→  });
    77→}

    83→async function preflightCheck(adapter: string, sourceKey: string): Promise<{ ok: boolean; cause: string }> {
    84→  // In a live deployment this would be three distinct probes. In the sandbox we
    85→  // simulate the structure — the cause labels are what matter (Design §6).
    86→  if (adapter === "X") {
    87→    return { ok: false, cause: "scraper rate-limited — backing off via 429 mitigation" };
    88→  }
    89→  return { ok: true, cause: "" };
    90→}

    92→// ─────────────────────────────────────────────────────────────────────
    93→// Checkpoint 3 — sampled extraction verification: verbatim quote string-matched
    94→// in stored raw. Failure quarantines the batch from all gate computation (L3).

    97→function locateInBody(body: string, quote: string): { start: number; end: number } | null {
    98→  const idx = body.indexOf(quote);
    99→  if (idx >= 0) return { start: idx, end: idx + quote.length };
   100→  // Fuzzy fallback — first 30 chars
   101→  const head = quote.slice(0, 30);
   102→  const fidx = body.indexOf(head);
   103→  if (fidx >= 0) return { start: fidx, end: fidx + quote.length };
   104→  return null;
   105→}

   107→// ─────────────────────────────────────────────────────────────────────
   108→// The two-pass pipeline: cheap TRIAGE → strong DEEP_EXTRACT above threshold
   109→// Content-hash cache: same hash + same extractionVersion never hits the LLM twice
   110→// ─────────────────────────────────────────────────────────────────────
   111→
   112→const triageSchema = z.object({
   113→  relevance: z.number(),
   114→  signal: z.enum(["ALPHA", "NOISE"]),
   115→  reason: z.string(),
   116→});
   117→
   118→const deepExtractSchema = z.object({
   119→  direction: z.enum(["BULLISH", "BEARISH", "NEUTRAL"]),
   120→  conviction: z.enum(["LOW", "MEDIUM", "HIGH"]),
   121→  insightType: z.enum(["FORECAST", "OBSERVATION", "OPINION"]),
   122→  verbatimQuote: z.string(),
   123→  keyInsight: z.string(),
   124→  tickers: z.array(z.string()).default([]),
   125→  entities: z.array(z.string()).default([]),
   126→  confidence: z.enum(["CLEAN", "HEDGED", "AMBIGUOUS"]),
   127→});

   129→async function triageAndExtract(bodyText: string, authorId: string, rawContentId: string) {
   130→  // Two-pass: cheap triage, then strong extraction only above threshold
   131→  const triage = await complete({
   132→    taskType: "TRIAGE",
   133→    prompt: { ...getPrompt("triage/v3"), params: { content: bodyText.slice(0, 1500) } },
   134→    schema: triageSchema,
   135→    cacheKey: hash(bodyText.slice(0, 500)),
   136→  });
   137→  const tri = triageSchema.safeParse(triage.data);
   138→  if (!tri.success || tri.data.relevance < 5 || tri.data.signal === "NOISE") {
   139→    return { skipped: true as const, reason: tri.success ? tri.data.reason : "parse-error" };
   140→  }
   141→
   142→  // Strong extraction
   143→  const ext = await complete({
   144→    taskType: "DEEP_EXTRACT",
   145→    prompt: { ...getPrompt("deep_extract/v3"), params: { content: bodyText.slice(0, 4000) } },
   146→    schema: deepExtractSchema,
   147→    cacheKey: hash(bodyText),
   148→  });
   149→  const ex = deepExtractSchema.safeParse(ext.data);
   150→  if (!ex.success) {
   151→    return { skipped: true as const, reason: "deep-extract parse-error" };
   152→  }
   153→
   154→  // Span anchor — locate the verbatim quote in the stored body
   155→  const span = locateInBody(bodyText, ex.data.verbatimQuote);
   156→  // L4: clamp dateLatest to fetchedAt
   157→  const fetchedAt = new Date();
   158→  const dateLatest = clampDateLatest(fetchedAt, fetchedAt);
   159→  if (dateLatest < fetchedAt) {
   160→    await logClamp({ rawContentId, field: "dateLatest", from: fetchedAt.toISOString(), to: dateLatest.toISOString() });
   161→  }
   162→
   163→  // CP3 — sample verification: the verbatim quote must be locatable in raw.
   164→  // If not, quarantine this source (L3: errors are never verdicts).
   165→  if (!span) {
   166→    return {
   167→      skipped: true as const,
   168→      reason: "CP3 violation — verbatim quote not found in stored raw; quarantined",
   169→      quarantined: true,
   170→    };
   171→  }
   172→
   173→  return {
   174→    skipped: false as const,
   175→    extracted: {
   176→      ...ex.data,
   177→      spanStart: span.start,
   178→      spanEnd: span.end,
   179→      spanConfidence: span ? "EXACT" : "FUZZY",
   180→      dateIso: fetchedAt,
   181→      dateEarliest: fetchedAt,
   182→      dateLatest,
   183→    },
   184→  };
   185→}

   191→async function storeRaw(args: {
   192→  url: string;
   193→  title: string;
   194→  bodyText: string;
   195→  adapterType: string;
   196→  adapterVersion: string;
   197→  threadId?: string;
   198→  referencesUrl?: string;
   199→  referenceType?: string;
   200→}): Promise<{ id: string; created: boolean }> {
   201→  const contentHash = hash(args.bodyText + args.url);
   202→  const existing = await db.rawContent.findUnique({ where: { contentHash } });
   203→  if (existing) return { id: existing.id, created: false };
   204→  const row = await db.rawContent.create({
   205→    data: {
   206→      contentHash,
   207→      url: args.url,
   208→      storageRef: `raw/${args.adapterType.toLowerCase()}/${contentHash}.txt`,
   209→      title: args.title,
   210→      adapterType: args.adapterType,
   211→      adapterVersion: args.adapterVersion,
   212→      bodyText: args.bodyText,
   213→      threadId: args.threadId,
   214→      referencesUrl: args.referencesUrl,
   215→      referenceType: args.referenceType,
   216→      fetchedAt: new Date(),
   217→      extractionStatus: "PENDING",
   218→    },
   219→  });
   220→  return { id: row.id, created: true };
   221→}

   234→export async function runRssAdapter() {
   235→  const jobRun = await startJobRun("adapters:rss");
   236→  const counts = { fetched: 0, new: 0, deduped: 0, extracted: 0, quarantined: 0, errors: 0 };
   237→
   238→  try {
   239→    for (const feed of RSS_FEEDS) {
   240→      const wm = await getWatermark("RSS", feed.feedUrl);
   241→      counts.fetched++;
   242→
   243→      // In production: fetch(feed.feedUrl), parse RSS XML, iterate <item> elements
   244→      // where item.guid > wm.lastGuid (or item.pubDate > wm.lastProcessedAt).
   245→      // Here we synthesize 0-1 new items per feed to demonstrate idempotency.
   246→      const newGuid = `substack:${feed.handle}:${Date.now()}`;
   247→      const alreadySeen = wm?.lastGuid === newGuid;
   248→
   249→      if (alreadySeen) {
   250→        counts.deduped++;
   251→        continue;
   252→      }
   253→
   254→      // Synthetic new item (in production this comes from the RSS parser)
   255→      const bodyText = `${feed.realName} — new post\n\nNo new channel checks today. Holding prior stance.`;
   256→      const { id: rawId, created } = await storeRaw({
   257→        url: `${feed.feedUrl}/${newGuid}`,
   258→        title: `${feed.realName} — daily update`,
   259→        bodyText,
   260→        adapterType: "RSS",
   261→        adapterVersion: "v1",
   262→      });
   263→
   264→      if (!created) {
   265→        counts.deduped++;
   266→      } else {
   267→        counts.new++;
   268→        // Author lookup
   269→        const author = await db.author.findUnique({ where: { handle: feed.handle } });
   270→        if (author) {
   271→          const result = await triageAndExtract(bodyText, author.id, rawId);
   272→          if (result.skipped) {
   273→            if ((result as any).quarantined) counts.quarantined++;
   274→            await db.rawContent.update({ where: { id: rawId }, data: { extractionStatus: "SKIPPED_TRIAGE" } });
   275→          } else {
   276→            counts.extracted++;
   277→            await db.source.create({
   278→              data: {
   279→                rawContentId: rawId,
   280→                extractionVersion: "deep_extract/v3",
   281→                authorId: author.id,
   282→                dateIso: result.extracted!.dateIso,
   283→                dateEarliest: result.extracted!.dateEarliest,
   284→                dateLatest: result.extracted!.dateLatest,
   285→                direction: result.extracted!.direction,
   286→                conviction: result.extracted!.conviction,
   287→                confidence: result.extracted!.confidence,
   288→                insightType: result.extracted!.insightType,
   289→                verbatimQuote: result.extracted!.verbatimQuote,
   290→                keyInsight: result.extracted!.keyInsight,
   291→                tickers: result.extracted!.tickers as any,
   292→                entities: result.extracted!.entities as any,
   293→                independenceClass: "UNCLASSIFIED",
   294→                spanStart: result.extracted!.spanStart,
   295→                spanEnd: result.extracted!.spanEnd,
   296→                spanConfidence: result.extracted!.spanConfidence,
   297→              },
   298→            });
   299→            await db.rawContent.update({ where: { id: rawId }, data: { extractionStatus: "EXTRACTED" } });
   300→          }
   301→        }
   302→      }
   303→
   304→      await setWatermark("RSS", feed.feedUrl, { lastGuid: newGuid });
   305→    }
   306→
   307→    await endJobRun(jobRun.id, "DONE", counts);
   308→    await updateAdapterHealth("rss", true);
   309→    return { ok: true, counts };
   310→  } catch (e: any) {
   311→    await endJobRun(jobRun.id, "FAILED", counts, e.message);
   312→    await updateAdapterHealth("rss", false, e.message);
   313→    return { ok: false, error: e.message, counts };
   314→  }
   315→}

   663→// ─────────────────────────────────────────────────────────────────────
   664→// pipeline:stance — per-event stance updates, change classification,
   665→// compound alerts (Spec §7, M5)
   666→// Exponential decay (~45d half-life), change classes CONSISTENT/MODERATING/
   667→// REVERSING/NEW_ENGAGEMENT/SILENCE
   668→// ─────────────────────────────────────────────────────────────────────
   669→
   670→const STANCE_DECAY_HALF_LIFE_DAYS = 45;
   671→
   672→function decayFactor(lastEventDate: Date, asOf: Date = new Date()): number {
   673→  const daysSince = (asOf.getTime() - lastEventDate.getTime()) / 86400_000;
   674→  if (daysSince <= 0) return 1.0;
   675→  return Math.pow(0.5, daysSince / STANCE_DECAY_HALF_LIFE_DAYS);
   676→}
   677→
   678→function classifyChange(prior: number, next: number, hasNewEngagement: boolean): string {
   679→  const delta = next - prior;
   680→  if (hasNewEngagement && prior === 0) return "NEW_ENGAGEMENT";
   681→  if (Math.abs(delta) < 0.05) return "CONSISTENT";
   682→  if (Math.abs(delta) < 0.2) return "MODERATING";
   683→  if (delta <= -0.2) return "REVERSING";
   684→  return "CONSISTENT";
   685→}

   687→export async function runStancePipeline() {
   688→  const jobRun = await startJobRun("pipeline:stance");
   689→  const counts = { updated: 0, changes: 0, alerts: 0, silence: 0 };
   690→
   691→  try {
   692→    const authors = await db.author.findMany({
   693→      include: { stances: true },
   694→    });
   695→
   696→    for (const author of authors) {
   697→      // Get this author's recent sources grouped by narrative family
   698→      const recentSources = await db.source.findMany({
   699→        where: {
   700→          authorId: author.id,
   701→          dateLatest: { gte: new Date(Date.now() - 60 * 86400_000) },
   702→        },
   703→        orderBy: { dateLatest: "desc" },
   704→        take: 20,
   705→      });
   706→
   707→      if (recentSources.length === 0) continue;
   708→
   709→      // Group by narrative family (from linked thesis via quantClaims, or from stance records)
   710→      for (const stance of author.stances) {
   711→        const familySources = recentSources.filter(s => {
   712→          // Match by entities or tickers to the family (simplified)
   713→          return s.direction !== "NEUTRAL";
   714→        });
   715→
   716→        if (familySources.length === 0) {
   717→          // Check silence — day-precision only (Spec §7)
   718→          if (stance.lastEventDate) {
   719→            const daysSince = (Date.now() - new Date(stance.lastEventDate).getTime()) / 86400_000;
   720→            if (daysSince > 30) {
   721→              counts.silence++;
   722→              await db.stanceChange.create({
   723→                data: {
   724→                  authorId: author.id,
   725→                  narrativeFamily: stance.narrativeFamily,
   726→                  changeType: "SILENCE",
   727→                  priorStance: stance.rollingDirection,
   728→                  newStance: stance.rollingDirection,
   729→                  magnitude: 0,
   730→                  reviewed: false,
   731→                },
   732→              });
   733→            }
   734→          }
   735→          continue;
   736→        }
   737→
   738→        // Compute new rolling stance with decay
   739→        const asOf = new Date();
   740→        let weightedSum = 0;
   741→        let weightSum = 0;
   742→        for (const s of familySources) {
   743→          const d = new Date(s.dateLatest ?? s.dateIso ?? asOf);
   744→          const w = decayFactor(d, asOf);
   745→          const dirVal = s.direction === "BULLISH" ? 1 : s.direction === "BEARISH" ? -1 : 0;
   746→          weightedSum += dirVal * w * (s.conviction === "HIGH" ? 1.0 : s.conviction === "MEDIUM" ? 0.7 : 0.4);
   747→          weightSum += w;
   748→        }
   749→        const newDirection = weightSum > 0 ? weightedSum / weightSum : 0;
   750→
   751→        // Apply book-talk discount (POSITIONED_MANAGER: 0.5× consistent, 1.5× on change)
   752→        let finalDirection = newDirection;
   753→        if (author.epistemicClass === "POSITIONED_MANAGER") {
   754→          const changed = Math.abs(newDirection - stance.rollingDirection) > 0.1;
   755→          finalDirection = newDirection * (changed ? 1.5 : 0.5);
   756→          finalDirection = Math.max(-1, Math.min(1, finalDirection));
   757→        }
   758→
   759→        const priorStance = stance.rollingDirection;
   760→        const changeType = classifyChange(priorStance, finalDirection, familySources.length > 0 && priorStance === 0);
   761→        const magnitude = Math.abs(finalDirection - priorStance);
   762→
   763→        // Update the stance
   764→        await db.authorStance.update({
   765→          where: { id: stance.id },
   766→          data: {
   767→            rollingDirection: Math.round(finalDirection * 100) / 100,
   768→            rollingConviction: Math.round((weightSum / familySources.length) * 100) / 100,
   769→            insightCount: stance.insightCount + familySources.length,
   770→            lastEventDate: new Date(familySources[0].dateLatest ?? asOf),
   771→          },
   772→        });
   773→        counts.updated++;
   774→
   775→        // Record change if significant
   776→        if (changeType !== "CONSISTENT" || magnitude > 0.05) {
   777→          counts.changes++;
   778→          const sc = await db.stanceChange.create({
   779→            data: {
   780→              authorId: author.id,
   781→              narrativeFamily: stance.narrativeFamily,
   782→              changeType,
   783→              priorStance,
   784→              newStance: finalDirection,
   785→              magnitude,
   786→              reviewed: false,
   787→            },
   788→          });
   789→
   790→          // Compound alert = stance change × upstream score × affected thesis stage
   791→          // (Spec §7 — "the system's single most actionable event class")
   792→          if (changeType === "REVERSING" || (changeType === "MODERATING" && magnitude > 0.15)) {
   793→            counts.alerts++;
   794→            await db.queueItem.create({
   795→              data: {
   796→                type: "ALERT",
   797→                priority: 1,
   798→                summary: `Compound stance alert: ${author.realName} ${changeType} on ${stance.narrativeFamily} (magnitude ${magnitude.toFixed(2)})`,
   799→                payload: {
   800→                  authorId: author.id,
   801→                  narrativeFamily: stance.narrativeFamily,
   802→                  changeType,
   803→                  magnitude,
   804→                  stanceChangeId: sc.id,
   805→                } as any,
   806→                status: "OPEN",
   807→              },
   808→            });
   809→          }
   810→        }
   811→      }
   812→    }
   813→
   814→    await endJobRun(jobRun.id, "DONE", counts);
   815→    return { ok: true, counts };
   816→  } catch (e: any) {
   817→    await endJobRun(jobRun.id, "FAILED", counts, e.message);
   818→    return { ok: false, error: e.message, counts };
   819→  }
   820→}

  1047→    const totalRaw = await db.rawContent.count();
  1048→    const discarded = await db.rawContent.count({ where: { extractionStatus: "SKIPPED_TRIAGE" } });
  1049→    counts.discard_rate = totalRaw > 0 ? Math.round((discarded / totalRaw) * 100) / 100 : 0;
  1050→
  1051→    // Verification pass rate
  1052→    const totalVerifications = await db.verificationEvent.count();
  1053→    const passedVerifications = await db.verificationEvent.count({
  1054→      where: { status: { startsWith: "PASSED" } },
  1055→    });
  1056→    counts.verification_pass_rate = totalVerifications > 0 ? Math.round((passedVerifications / totalVerifications) * 100) : 0;
  1057→
  1058→    // Queue latency (average hours items sat OPEN)
  1059→    const openItems = await db.queueItem.findMany({ where: { status: "OPEN" } });
  1060→    if (openItems.length > 0) {
  1061→      const totalLatency = openItems.reduce((s, q) => s + (Date.now() - new Date(q.createdAt).getTime()) / 3600_000, 0);
  1062→      counts.queue_latency_hrs = Math.round((totalLatency / openItems.length) * 10) / 10;
  1063→    }
  1064→
  1065→    // Attribution flags (carrier ≠ speaker cases)
  1066→    counts.attribution_flags = await db.source.count({ where: { carrierAuthorId: { not: null } } });
  1067→
  1068→    await endJobRun(jobRun.id, "DONE", counts);
  1069→    return { ok: true, counts };
  1070→  } catch (e: any) {
  1071→    await endJobRun(jobRun.id, "FAILED", counts, e.message);
  1072→    return { ok: false, error: e.message, counts };
  1073→  }
  1074→}
