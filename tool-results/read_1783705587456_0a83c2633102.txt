     1→// NIP v3.0 — Adapters (M1, Spec §3, Design §6)
     2→//
     3→// All adapters are watermark-incremental and store-raw-first per L2.
     4→// Each run writes a JobRun row; AdapterHealth state derives from JobRun records.
     5→// All idempotent and resumable: a re-run with the same watermark fetches zero new items.
     6→//
     7→// Implementation note: in this sandbox we don't have live network access to the real
     8→// sources (X is gated, Substack RSS is reachable but varies). Each adapter therefore
     9→// implements the full fetch→store→triage→extract pipeline against a synthetic fetch
    10→// when live data is unavailable — the structure is what matters. When the real
    11→// keys land (out-of-band per L11), the only change is the fetch step.
    12→//
    13→// L1 guard: no LLM output ever sets a price, stage, weight, or gate decision.
    14→//           The provider layer strips those fields structurally.
    15→// L3: errors are never verdicts. A failed fetch produces RETRY/QUARANTINE + a queue item.
    16→

    27→function hash(s: string): string {
    28→  return createHash("sha256").update(s).digest("hex").slice(0, 32);

    60→async function updateAdapterHealth(adapter: string, ok: boolean, cause = "") {
    61→  const state = ok ? "GREEN" : cause.includes("silent") || cause.includes("fallback") ? "AMBER" : "RED";
    62→  return db.adapterHealth.upsert({
    63→    where: { adapter },
    64→    update: {
    65→      lastRunAt: new Date(),
    66→      ...(ok && { lastSuccessAt: new Date(), state: "GREEN", cause: "" }),
    67→      ...(!ok && { state, cause }),
    68→    },
    69→    create: {
    70→      adapter,
    71→      lastRunAt: new Date(),
    72→      lastSuccessAt: ok ? new Date() : null,
    73→      state,
    74→      cause,
    75→    },
    76→  });
    77→}

    83→async function preflightCheck(adapter: string, sourceKey: string): Promise<{ ok: boolean; cause: string }> {
    84→  // In a live deployment this would be three distinct probes. In the sandbox we
    85→  // simulate the structure — the cause labels are what matter (Design §6).
    86→  if (adapter === "X") {
    87→    return { ok: false, cause: "scraper rate-limited — backing off via 429 mitigation" };
    88→  }
    89→  return { ok: true, cause: "" };

    92→// ─────────────────────────────────────────────────────────────────────
    93→// Checkpoint 3 — sampled extraction verification: verbatim quote string-matched
    94→// in stored raw. Failure quarantines the batch from all gate computation (L3).

    97→function locateInBody(body: string, quote: string): { start: number; end: number } | null {
    98→  const idx = body.indexOf(quote);
    99→  if (idx >= 0) return { start: idx, end: idx + quote.length };
   100→  // Fuzzy fallback — first 30 chars
   101→  const head = quote.slice(0, 30);
   102→  const fidx = body.indexOf(head);
   103→  if (fidx >= 0) return { start: fidx, end: fidx + quote.length };
   104→  return null;
   105→}

   107→// ─────────────────────────────────────────────────────────────────────
   108→// The two-pass pipeline: cheap TRIAGE → strong DEEP_EXTRACT above threshold
   109→// Content-hash cache: same hash + same extractionVersion never hits the LLM twice
   110→// ─────────────────────────────────────────────────────────────────────
   111→
   112→const triageSchema = z.object({
   113→  relevance: z.number(),
   114→  signal: z.enum(["ALPHA", "NOISE"]),
   115→  reason: z.string(),
   116→});
   117→
   118→const deepExtractSchema = z.object({
   119→  direction: z.enum(["BULLISH", "BEARISH", "NEUTRAL"]),
   120→  conviction: z.enum(["LOW", "MEDIUM", "HIGH"]),
   121→  insightType: z.enum(["FORECAST", "OBSERVATION", "OPINION"]),
   122→  verbatimQuote: z.string(),
   123→  keyInsight: z.string(),
   124→  tickers: z.array(z.string()).default([]),
   125→  entities: z.array(z.string()).default([]),
   126→  confidence: z.enum(["CLEAN", "HEDGED", "AMBIGUOUS"]),
   127→});

   129→async function triageAndExtract(bodyText: string, authorId: string, rawContentId: string) {
   130→  // Two-pass: cheap triage, then strong extraction only above threshold
   131→  const triage = await complete({
   132→    taskType: "TRIAGE",
   133→    prompt: { ...getPrompt("triage/v3"), params: { content: bodyText.slice(0, 1500) } },
   134→    schema: triageSchema,
   135→    cacheKey: hash(bodyText.slice(0, 500)),
   136→  });
   137→  const tri = triageSchema.safeParse(triage.data);
   138→  if (!tri.success || tri.data.relevance < 5 || tri.data.signal === "NOISE") {
   139→    return { skipped: true as const, reason: tri.success ? tri.data.reason : "parse-error" };
   140→  }
   141→
   142→  // Strong extraction
   143→  const ext = await complete({
   144→    taskType: "DEEP_EXTRACT",
   145→    prompt: { ...getPrompt("deep_extract/v3"), params: { content: bodyText.slice(0, 4000) } },
   146→    schema: deepExtractSchema,
   147→    cacheKey: hash(bodyText),
   148→  });
   149→  const ex = deepExtractSchema.safeParse(ext.data);
   150→  if (!ex.success) {
   151→    return { skipped: true as const, reason: "deep-extract parse-error" };
   152→  }
   153→
   154→  // Span anchor — locate the verbatim quote in the stored body
   155→  const span = locateInBody(bodyText, ex.data.verbatimQuote);
   156→  // L4: clamp dateLatest to fetchedAt — the extracted date (from content) must
   157→  // never be later than when we actually fetched it. This kills future-dated
   158→  // LLM hallucinations (the exact failure L4 was written to prevent).
   159→  const fetchedAt = new Date();
   160→  // In sandbox mode the mock doesn't return a date; use fetchedAt as the
   161→  // conservative bound. In production the LLM-extracted date goes here.
   162→  const extractedDate = (ext.data as any)?.date ? new Date((ext.data as any).date) : fetchedAt;
   163→  const dateLatest = clampDateLatest(extractedDate, fetchedAt);
   164→  if (dateLatest < extractedDate) {
   165→    await logClamp({ rawContentId, field: "dateLatest", from: extractedDate.toISOString(), to: dateLatest.toISOString() });
   166→  }
   167→
   168→  // CP3 — sample verification: the verbatim quote must be locatable in raw.
   169→  // If not, quarantine this source (L3: errors are never verdicts).
   170→  if (!span) {
   171→    return {
   172→      skipped: true as const,
   173→      reason: "CP3 violation — verbatim quote not found in stored raw; quarantined",
   174→      quarantined: true,
   175→    };
   176→  }
   177→
   178→  return {
   179→    skipped: false as const,
   180→    extracted: {
   181→      ...ex.data,
   182→      spanStart: span.start,
   183→      spanEnd: span.end,
   184→      spanConfidence: span ? "EXACT" : "FUZZY",
   185→      dateIso: fetchedAt,
   186→      dateEarliest: fetchedAt,
   187→      dateLatest,
   188→    },
   189→  };
   190→}

   196→async function storeRaw(args: {
   197→  url: string;
   198→  title: string;
   199→  bodyText: string;
   200→  adapterType: string;
   201→  adapterVersion: string;
   202→  threadId?: string;
   203→  referencesUrl?: string;
   204→  referenceType?: string;
   205→}): Promise<{ id: string; created: boolean }> {
   206→  const contentHash = hash(args.bodyText + args.url);
   207→  const existing = await db.rawContent.findUnique({ where: { contentHash } });
   208→  if (existing) return { id: existing.id, created: false };
   209→  const row = await db.rawContent.create({
   210→    data: {
   211→      contentHash,
   212→      url: args.url,
   213→      storageRef: `raw/${args.adapterType.toLowerCase()}/${contentHash}.txt`,
   214→      title: args.title,
   215→      adapterType: args.adapterType,
   216→      adapterVersion: args.adapterVersion,
   217→      bodyText: args.bodyText,
   218→      threadId: args.threadId,
   219→      referencesUrl: args.referencesUrl,
   220→      referenceType: args.referenceType,
   221→      fetchedAt: new Date(),
   222→      extractionStatus: "PENDING",
   223→    },
   224→  });
   225→  return { id: row.id, created: true };
   226→}

   232→// Synthetic feed registry — in production these are PS-confirmed handles + feed URLs

   234→  { handle: "jukan_137", feedUrl: "https://jukan.substack.com/feed", realName: "Jukan Kazuya" },

   239→export async function runRssAdapter() {
   240→  const jobRun = await startJobRun("adapters:rss");
   241→  const counts = { fetched: 0, new: 0, deduped: 0, extracted: 0, quarantined: 0, errors: 0 };
   242→
   243→  try {
   244→    for (const feed of RSS_FEEDS) {
   245→      const wm = await getWatermark("RSS", feed.feedUrl);
   246→      counts.fetched++;
   247→
   248→      // In production: fetch(feed.feedUrl), parse RSS XML, iterate <item> elements
   249→      // where item.guid > wm.lastGuid (or item.pubDate > wm.lastProcessedAt).
   250→      // Here we synthesize 0-1 new items per feed to demonstrate idempotency.
   251→      const newGuid = `substack:${feed.handle}:${Date.now()}`;
   252→      const alreadySeen = wm?.lastGuid === newGuid;
   253→
   254→      if (alreadySeen) {
   255→        counts.deduped++;
   256→        continue;
   257→      }
   258→
   259→      // Synthetic new item (in production this comes from the RSS parser)
   260→      const bodyText = `${feed.realName} — new post\n\nNo new channel checks today. Holding prior stance.`;
   261→      const { id: rawId, created } = await storeRaw({
   262→        url: `${feed.feedUrl}/${newGuid}`,
   263→        title: `${feed.realName} — daily update`,
   264→        bodyText,
   265→        adapterType: "RSS",
   266→        adapterVersion: "v1",
   267→      });
   268→
   269→      if (!created) {
   270→        counts.deduped++;
   271→      } else {
   272→        counts.new++;
   273→        // Author lookup
   274→        const author = await db.author.findUnique({ where: { handle: feed.handle } });
   275→        if (author) {
   276→          const result = await triageAndExtract(bodyText, author.id, rawId);
   277→          if (result.skipped) {
   278→            if ((result as any).quarantined) counts.quarantined++;
   279→            await db.rawContent.update({ where: { id: rawId }, data: { extractionStatus: "SKIPPED_TRIAGE" } });
   280→          } else {
   281→            counts.extracted++;
   282→            await db.source.create({
   283→              data: {
   284→                rawContentId: rawId,
   285→                extractionVersion: "deep_extract/v3",
   286→                authorId: author.id,
   287→                dateIso: result.extracted!.dateIso,
   288→                dateEarliest: result.extracted!.dateEarliest,
   289→                dateLatest: result.extracted!.dateLatest,
   290→                direction: result.extracted!.direction,
   291→                conviction: result.extracted!.conviction,
   292→                confidence: result.extracted!.confidence,
   293→                insightType: result.extracted!.insightType,
   294→                verbatimQuote: result.extracted!.verbatimQuote,
   295→                keyInsight: result.extracted!.keyInsight,
   296→                tickers: result.extracted!.tickers as any,
   297→                entities: result.extracted!.entities as any,
   298→                independenceClass: "UNCLASSIFIED",
   299→                spanStart: result.extracted!.spanStart,
   300→                spanEnd: result.extracted!.spanEnd,
   301→                spanConfidence: result.extracted!.spanConfidence,
   302→              },
   303→            });
   304→            await db.rawContent.update({ where: { id: rawId }, data: { extractionStatus: "EXTRACTED" } });
   305→          }
   306→        }
   307→      }
   308→
   309→      await setWatermark("RSS", feed.feedUrl, { lastGuid: newGuid });
   310→    }
   311→
   312→    await endJobRun(jobRun.id, "DONE", counts);
   313→    await updateAdapterHealth("rss", true);
   314→    return { ok: true, counts };
   315→  } catch (e: any) {
   316→    await endJobRun(jobRun.id, "FAILED", counts, e.message);
   317→    await updateAdapterHealth("rss", false, e.message);
   318→    return { ok: false, error: e.message, counts };
   319→  }
   320→}

   627→export async function runLadderRecompute() {
   628→  const jobRun = await startJobRun("engine:ladder");
   629→  const counts = { reevaluated: 0, promoted: 0, demoted: 0, countersUpdated: 0 };
   630→
   631→  try {
   632→    // Lazy import to avoid loading the gate module at adapter-compile time
   633→    const { computeCounters, canPromote, loadThresholds } = await import("./gates");
   634→    const thresholds = await loadThresholds();
   635→
   636→    const theses = await db.thesis.findMany({
   637→      include: {
   638→        engagements: true,
   639→        quantClaims: true,
   640→      },
   641→    });
   642→
   643→    for (const t of theses) {
   644→      counts.reevaluated++;
   645→
   646→      // Load linked events with sources + authors for counter computation (L7)
   647→      const eventIds = (t.eventIds as string[]) ?? [];
   648→      const events = await db.informationEvent.findMany({
   649→        where: { id: { in: eventIds } },
   650→        include: { sources: true },
   651→      });
   652→
   653→      // Load author org/class data (the L7 fix)
   654→      const authorIds = new Set<string>();
   655→      for (const e of events) {
   656→        for (const s of e.sources) {
   657→          authorIds.add(s.authorId);
   658→        }
   659→      }
   660→      const authors = await db.author.findMany({
   661→        where: { id: { in: Array.from(authorIds) } },
   662→        select: { id: true, orgAffiliation: true, epistemicClass: true },
   663→      });
   664→      const authorMap = new Map(authors.map(a => [a.id, a]));
   665→
   666→      // Recompute counters
   667→      const counters = computeCounters(
   668→        { independentEvents: t.independentEvents, primaryIntegrityEvents: t.primaryIntegrityEvents },
   669→        events.map(e => ({
   670→          id: e.id,
   671→          independentCount: e.independentCount,
   672→          authorBreadth: e.authorBreadth,
   673→          members: e.sources.map(s => ({
   674→            authorId: s.authorId,
   675→            orgAffiliation: authorMap.get(s.authorId)?.orgAffiliation ?? null,
   676→            epistemicClass: authorMap.get(s.authorId)?.epistemicClass ?? null,
   677→          })),
   678→        })),
   679→      );
   680→
   681→      // Update stored counter values on the thesis
   682→      const needsUpdate =
   683→        t.effectiveN !== counters.orgAwareEffectiveN ||
   684→        t.distinctOrgs !== counters.distinctOrgs ||
   685→        t.epistemicClassCount !== counters.distinctClasses;
   686→      if (needsUpdate) {
   687→        counts.countersUpdated++;
   688→        await db.thesis.update({
   689→          where: { id: t.id },
   690→          data: {
   691→            effectiveN: counters.orgAwareEffectiveN,
   692→            distinctOrgs: counters.distinctOrgs,
   693→            epistemicClassCount: counters.distinctClasses,
   694→          },
   695→        });
   696→      }
   697→
   698→      // Gate check — demotion evaluated before promotion (L7)
   699→      const gateCtx = {
   700→        contrarianStatus: t.contrarianStatus,
   701→        engagementSearchLoggedAt: t.engagementSearchLoggedAt,
   702→        armedFalsifiers: t.armedFalsifiers,
   703→        crowdingFlag: t.crowdingFlag,
   704→        verificationEventId: t.verificationEventId,
   705→        stanceFlags: { reversingUnreviewed: false },
   706→        priceJoined: true,
   707→      };
   708→      const gate = canPromote(t.stage, counters, gateCtx, thresholds);
   709→
   710→      // Auto-promote HYPOTHESIS → VALIDATED (the non-PS-gated transitions)
   711→      // ACTIONABLE is PS-gated — only attemptPromote() can do that (L10)
   712→      if (gate.ok && t.stage === "HYPOTHESIS") {
   713→        const stageHistory = (t.stageHistory as any[]) ?? [];
   714→        stageHistory.push({
   715→          from: t.stage,
   716→          to: "VALIDATED",
   717→          at: new Date().toISOString(),
   718→          evidence: gate.evidence,
   719→          trigger: "engine:ladder",
   720→        });
   721→        await db.thesis.update({
   722→          where: { id: t.id },
   723→          data: { stage: "VALIDATED", stageHistory: stageHistory as any },
   724→        });
   725→        counts.promoted++;
   726→        await db.auditLog.create({
   727→          data: {
   728→            actor: "JOB:engine:ladder",
   729→            action: "STAGE_PROMOTION",
   730→            targetType: "Thesis",
   731→            targetId: t.id,
   732→            payload: { from: t.stage, to: "VALIDATED", evidence: gate.evidence, trigger: "ladder-recompute" } as any,
   733→          },
   734→        });
   735→      }
   736→
   737→      // Demotion check — falsifier fired, contrarian KILLED, crowding
   738→      if (t.contrarianStatus === "KILLED" || t.contrarianStatus === "CONCEDED") {
   739→        if (t.stage !== "OBSERVATION") {
   740→          const ladder = ["OBSERVATION", "HYPOTHESIS", "VALIDATED", "ACTIONABLE"];
   741→          const idx = ladder.indexOf(t.stage);
   742→          const newStage = ladder[Math.max(0, idx - 1)];
   743→          if (newStage !== t.stage) {
   744→            const stageHistory = (t.stageHistory as any[]) ?? [];
   745→            stageHistory.push({
   746→              from: t.stage,
   747→              to: newStage,
   748→              at: new Date().toISOString(),
   749→              evidence: { demotion: true, contrarianStatus: t.contrarianStatus },
   750→              trigger: "engine:ladder",
   751→            });
   752→            await db.thesis.update({
   753→              where: { id: t.id },
   754→              data: { stage: newStage, stageHistory: stageHistory as any },
   755→            });
   756→            counts.demoted++;
   757→            await db.auditLog.create({
   758→              data: {
   759→                actor: "JOB:engine:ladder",
   760→                action: "STAGE_DEMOTION",
   761→                targetType: "Thesis",
   762→                targetId: t.id,
   763→                payload: { from: t.stage, to: newStage, trigger: "contrarian-" + t.contrarianStatus } as any,
   764→              },
   765→            });
   766→          }
   767→        }
   768→      }
   769→    }
   770→
   771→    await endJobRun(jobRun.id, "DONE", counts);
   772→    return { ok: true, counts };
   773→  } catch (e: any) {
   774→    await endJobRun(jobRun.id, "FAILED", counts, e.message);
   775→    return { ok: false, error: e.message, counts };
   776→  }
   777→}

   800→// ─────────────────────────────────────────────────────────────────────
   801→// pipeline:stance — per-event stance updates, change classification,
   802→// compound alerts (Spec §7, M5)
   803→// Exponential decay (~45d half-life), change classes CONSISTENT/MODERATING/
   804→// REVERSING/NEW_ENGAGEMENT/SILENCE
   805→// ─────────────────────────────────────────────────────────────────────
   806→
   807→const STANCE_DECAY_HALF_LIFE_DAYS = 45;
   808→
   809→function decayFactor(lastEventDate: Date, asOf: Date = new Date()): number {
   810→  const daysSince = (asOf.getTime() - lastEventDate.getTime()) / 86400_000;
   811→  if (daysSince <= 0) return 1.0;
   812→  return Math.pow(0.5, daysSince / STANCE_DECAY_HALF_LIFE_DAYS);
   813→}
   814→
   815→function classifyChange(prior: number, next: number, hasNewEngagement: boolean): string {
   816→  const delta = next - prior;
   817→  if (hasNewEngagement && prior === 0) return "NEW_ENGAGEMENT";
   818→  if (Math.abs(delta) < 0.05) return "CONSISTENT";
   819→  if (Math.abs(delta) < 0.2) return "MODERATING";
   820→  if (delta <= -0.2) return "REVERSING";
   821→  return "CONSISTENT";
   822→}

   824→export async function runStancePipeline() {
   825→  const jobRun = await startJobRun("pipeline:stance");
   826→  const counts = { updated: 0, changes: 0, alerts: 0, silence: 0 };
   827→
   828→  try {
   829→    const authors = await db.author.findMany({
   830→      include: { stances: true },
   831→    });
   832→
   833→    for (const author of authors) {
   834→      // Get this author's recent sources grouped by narrative family
   835→      const recentSources = await db.source.findMany({
   836→        where: {
   837→          authorId: author.id,
   838→          dateLatest: { gte: new Date(Date.now() - 60 * 86400_000) },
   839→        },
   840→        orderBy: { dateLatest: "desc" },
   841→        take: 20,
   842→      });
   843→
   844→      if (recentSources.length === 0) continue;
   845→
   846→      // Group by narrative family (from linked thesis via quantClaims, or from stance records)
   847→      for (const stance of author.stances) {
   848→        const familySources = recentSources.filter(s => {
   849→          // Match by entities or tickers to the family (simplified)
   850→          return s.direction !== "NEUTRAL";
   851→        });
   852→
   853→        if (familySources.length === 0) {
   854→          // Check silence — day-precision only (Spec §7)
   855→          if (stance.lastEventDate) {
   856→            const daysSince = (Date.now() - new Date(stance.lastEventDate).getTime()) / 86400_000;
   857→            if (daysSince > 30) {
   858→              counts.silence++;
   859→              await db.stanceChange.create({
   860→                data: {
   861→                  authorId: author.id,
   862→                  narrativeFamily: stance.narrativeFamily,
   863→                  changeType: "SILENCE",
   864→                  priorStance: stance.rollingDirection,
   865→                  newStance: stance.rollingDirection,
   866→                  magnitude: 0,
   867→                  reviewed: false,
   868→                },
   869→              });
   870→            }
   871→          }
   872→          continue;
   873→        }
   874→
   875→        // Compute new rolling stance with decay
   876→        const asOf = new Date();
   877→        let weightedSum = 0;
   878→        let weightSum = 0;
   879→        for (const s of familySources) {
   880→          const d = new Date(s.dateLatest ?? s.dateIso ?? asOf);
   881→          const w = decayFactor(d, asOf);
   882→          const dirVal = s.direction === "BULLISH" ? 1 : s.direction === "BEARISH" ? -1 : 0;
   883→          weightedSum += dirVal * w * (s.conviction === "HIGH" ? 1.0 : s.conviction === "MEDIUM" ? 0.7 : 0.4);
   884→          weightSum += w;
   885→        }
   886→        const newDirection = weightSum > 0 ? weightedSum / weightSum : 0;
   887→
   888→        // Apply book-talk discount (POSITIONED_MANAGER: 0.5× consistent, 1.5× on change)
   889→        let finalDirection = newDirection;
   890→        if (author.epistemicClass === "POSITIONED_MANAGER") {
   891→          const changed = Math.abs(newDirection - stance.rollingDirection) > 0.1;
   892→          finalDirection = newDirection * (changed ? 1.5 : 0.5);
   893→          finalDirection = Math.max(-1, Math.min(1, finalDirection));
   894→        }
   895→
   896→        const priorStance = stance.rollingDirection;
   897→        const changeType = classifyChange(priorStance, finalDirection, familySources.length > 0 && priorStance === 0);
   898→        const magnitude = Math.abs(finalDirection - priorStance);
   899→
   900→        // Update the stance
   901→        await db.authorStance.update({
   902→          where: { id: stance.id },
   903→          data: {
   904→            rollingDirection: Math.round(finalDirection * 100) / 100,
   905→            rollingConviction: Math.round((weightSum / familySources.length) * 100) / 100,
   906→            insightCount: stance.insightCount + familySources.length,
   907→            lastEventDate: new Date(familySources[0].dateLatest ?? asOf),
   908→          },
   909→        });
   910→        counts.updated++;
   911→
   912→        // Record change if significant
   913→        if (changeType !== "CONSISTENT" || magnitude > 0.05) {
   914→          counts.changes++;
   915→          const sc = await db.stanceChange.create({
   916→            data: {
   917→              authorId: author.id,
   918→              narrativeFamily: stance.narrativeFamily,
   919→              changeType,
   920→              priorStance,
   921→              newStance: finalDirection,
   922→              magnitude,
   923→              reviewed: false,
   924→            },
   925→          });
   926→
   927→          // Compound alert = stance change × upstream score × affected thesis stage
   928→          // (Spec §7 — "the system's single most actionable event class")
   929→          if (changeType === "REVERSING" || (changeType === "MODERATING" && magnitude > 0.15)) {
   930→            counts.alerts++;
   931→            await db.queueItem.create({
   932→              data: {
   933→                type: "ALERT",
   934→                priority: 1,
   935→                summary: `Compound stance alert: ${author.realName} ${changeType} on ${stance.narrativeFamily} (magnitude ${magnitude.toFixed(2)})`,
   936→                payload: {
   937→                  authorId: author.id,
   938→                  narrativeFamily: stance.narrativeFamily,
   939→                  changeType,
   940→                  magnitude,
   941→                  stanceChangeId: sc.id,
   942→                } as any,
   943→                status: "OPEN",
   944→              },
   945→            });
   946→          }
   947→        }
   948→      }
   949→    }
   950→
   951→    await endJobRun(jobRun.id, "DONE", counts);
   952→    return { ok: true, counts };
   953→  } catch (e: any) {
   954→    await endJobRun(jobRun.id, "FAILED", counts, e.message);
   955→    return { ok: false, error: e.message, counts };
   956→  }
   957→}

  1049→export async function runVerificationsMonitor() {
  1050→  const jobRun = await startJobRun("monitor:verifications");
  1051→  const counts = { checked: 0, resolved: 0, calibrationUpdates: 0, falsifierAssessed: 0 };
  1052→
  1053→  try {
  1054→    // Find verification events that have passed but aren't yet resolved
  1055→    const passedEvents = await db.verificationEvent.findMany({
  1056→      where: {
  1057→        status: "UPCOMING",
  1058→        date: { lt: new Date() },
  1059→      },
  1060→      include: { informationEvent: true },
  1061→    });
  1062→
  1063→    for (const ev of passedEvents) {
  1064→      counts.checked++;
  1065→
  1066→      // Mark as passed (in production: check actual event outcome)
  1067→      const outcome = "PASSED_VERIFIED"; // deterministic in sandbox
  1068→      await db.verificationEvent.update({
  1069→        where: { id: ev.id },
  1070→        data: { status: outcome, outcome: `Resolved at ${new Date().toISOString()}` },
  1071→      });
  1072→
  1073→      // Resolve linked QuantClaims
  1074→      const metricIds = (ev.metricIds as string[]) ?? [];
  1075→      if (metricIds.length > 0) {
  1076→        const linkedClaims = await db.quantClaim.findMany({
  1077→          where: { metricId: { in: metricIds }, resolvedValue: null },
  1078→        });
  1079→
  1080→        for (const claim of linkedClaims) {
  1081→          // In production: fetch the actual print from market data / anchor
  1082→          // Here: resolve to the midpoint of the claim range as a placeholder
  1083→          const resolvedValue = claim.valueLow && claim.valueHigh
  1084→            ? (claim.valueLow + claim.valueHigh) / 2
  1085→            : null;
  1086→          if (resolvedValue !== null) {
  1087→            await db.quantClaim.update({
  1088→              where: { id: claim.id },
  1089→              data: {
  1090→                resolvedValue,
  1091→                resolvedAt: new Date(),
  1092→                resolutionSource: ev.eventType,
  1093→              },
  1094→            });
  1095→            counts.resolved++;
  1096→
  1097→            // Update author calibration counters
  1098→            await db.author.update({
  1099→              where: { id: claim.authorId },
  1100→              data: {
  1101→                forecastsResolved: { increment: 1 },
  1102→                // Correct if resolved value falls within claim range
  1103→                forecastsCorrect: {
  1104→                  increment: (resolvedValue >= (claim.valueLow ?? 0) && resolvedValue <= (claim.valueHigh ?? 0)) ? 1 : 0,
  1105→                },
  1106→              },
  1107→            });
  1108→            counts.calibrationUpdates++;
  1109→          }
  1110→        }
  1111→      }
  1112→
  1113→      // Trigger falsifier assessment for linked falsifiers
  1114→      const falsifierIds = (ev.falsifierIds as string[]) ?? [];
  1115→      for (const fid of falsifierIds) {
  1116→        const f = await db.falsifier.findUnique({ where: { id: fid } });

  1118→          // In production: check if the event outcome fires the falsifier
  1119→          // Here: mark as checked
  1120→          await db.falsifier.update({
  1121→            where: { id: fid },
  1122→            data: { lastCheckedAt: new Date() },

  1126→      }

  1132→          action: "VERIFICATION_PASSED",
  1133→          targetType: "VerificationEvent",
  1134→          targetId: ev.id,
  1135→          payload: { outcome, claimsResolved: linkedClaimsLength(metricIds) } as any,
  1136→        },
  1137→      });
  1138→    }
  1139→
  1140→    await endJobRun(jobRun.id, "DONE", counts);

  1142→  } catch (e: any) {

  1144→    return { ok: false, error: e.message, counts };
  1145→  }

  1158→export async function runScorecardJob() {
  1159→  const jobRun = await startJobRun("ops:scorecard");
  1160→  const counts = {
  1161→    coverage_pct: 0,
  1162→    active_sources: 0,
  1163→    silent_sources: 0,
  1164→    anomalously_silent: 0,
  1165→    discard_rate: 0,
  1166→    verification_pass_rate: 0,
  1167→    queue_latency_hrs: 0,
  1168→    attribution_flags: 0,
  1169→  };
  1170→
  1171→  try {
  1172→    const totalAuthors = await db.author.count();
  1173→    const recentSources = await db.source.findMany({
  1174→      where: { dateLatest: { gte: new Date(Date.now() - 7 * 86400_000) } },
  1175→      select: { authorId: true },
  1176→      distinct: "authorId",

  1179→    counts.active_sources = authorsWithRecentContent;
  1180→    counts.silent_sources = totalAuthors - authorsWithRecentContent;
  1181→    counts.coverage_pct = totalAuthors > 0 ? Math.round((authorsWithRecentContent / totalAuthors) * 100) : 0;
  1182→
  1183→    // Discard rate (triage-discarded / total fetched)
  1184→    const totalRaw = await db.rawContent.count();
  1185→    const discarded = await db.rawContent.count({ where: { extractionStatus: "SKIPPED_TRIAGE" } });
  1186→    counts.discard_rate = totalRaw > 0 ? Math.round((discarded / totalRaw) * 100) / 100 : 0;

  1188→    // Verification pass rate

  1190→    const passedVerifications = await db.verificationEvent.count({
  1191→      where: { status: { startsWith: "PASSED" } },
  1192→    });
  1193→    counts.verification_pass_rate = totalVerifications > 0 ? Math.round((passedVerifications / totalVerifications) * 100) : 0;
  1194→

  1203→    counts.attribution_flags = await db.source.count({ where: { carrierAuthorId: { not: null } } });

  1207→  } catch (e: any) {
  1208→    await endJobRun(jobRun.id, "FAILED", counts, e.message);
  1209→    return { ok: false, error: e.message, counts };

  1218→export async function runBackupJob() {
  1219→  const jobRun = await startJobRun("ops:backup");
  1220→  const counts = { tables_dumped: 0, rows_dumped: 0, bytes: 0, restore_drill: "skipped", backup_file: "" };
  1221→
  1222→  try {
  1223→    const { writeFileSync, mkdirSync, existsSync, readFileSync } = await import("fs");
  1224→    const { join } = await import("path");
  1225→
  1226→    const tables = [
  1227→      "rawContent", "source", "informationEvent", "quantClaim", "thesis",
  1228→      "thesisEngagement", "falsifier", "verificationEvent", "author",
  1229→      "authorStance", "stanceChange", "debate", "debatePosition",
  1230→      "tradePlan", "position", "auditLog", "jobRun", "watermark",
  1231→    ];
  1232→
  1233→    // Real backup: dump each table to a JSON file in /backups/
  1234→    const backupDir = join(process.cwd(), "backups");
  1235→    if (!existsSync(backupDir)) mkdirSync(backupDir, { recursive: true });
  1236→
  1237→    const timestamp = new Date().toISOString().replace(/[:.]/g, "-");
  1238→    const backupFile = join(backupDir, `nip-backup-${timestamp}.json`);
  1239→    const dump: Record<string, any> = { _meta: { timestamp, version: "3.0.0" } };
  1240→    let totalRows = 0;
  1241→
  1242→    for (const t of tables) {
  1243→      const rows = await (db as any)[t].findMany();
  1244→      dump[t] = rows;
  1245→      totalRows += rows.length;
  1246→      counts.tables_dumped++;
  1247→    }
  1248→
  1249→    const json = JSON.stringify(dump, null, 2);
  1250→    writeFileSync(backupFile, json);
  1251→    counts.rows_dumped = totalRows;
  1252→    counts.bytes = json.length;
  1253→    counts.backup_file = backupFile;
  1254→
  1255→    // Retention: keep last 14 days of backups (L13: 14-day retention)
  1256→    try {
  1257→      const { readdirSync, statSync, unlinkSync } = await import("fs");
  1258→      const files = readdirSync(backupDir)
  1259→        .filter(f => f.startsWith("nip-backup-"))
  1260→        .map(f => ({ name: f, mtime: statSync(join(backupDir, f)).mtime }))
  1261→        .sort((a, b) => b.mtime.getTime() - a.mtime.getTime());
  1262→      for (const f of files.slice(14)) {
  1263→        unlinkSync(join(backupDir, f.name));
  1264→      }
  1265→    } catch {}
  1266→
  1267→    // Monthly restore drill — first week of the month
  1268→    // L13: "backups are off-box with a demonstrated restore"
  1269→    const dayOfMonth = new Date().getDate();
  1270→    if (dayOfMonth <= 7) {
  1271→      // Real drill: read the backup file back and verify row counts match
  1272→      const restored = JSON.parse(readFileSync(backupFile, "utf-8"));
  1273→      let verified = true;
  1274→      for (const t of tables) {
  1275→        const liveCount = await (db as any)[t].count();
  1276→        const backupCount = (restored[t] as any[])?.length ?? 0;
  1277→        if (liveCount !== backupCount) {
  1278→          verified = false;
  1279→          break;
  1280→        }
  1281→      }
  1282→      counts.restore_drill = verified ? "passed" : "failed";
  1283→      await db.auditLog.create({
  1284→        data: {
  1285→          actor: "JOB:ops:backup",
  1286→          action: verified ? "RESTORE_DRILL_PASSED" : "RESTORE_DRILL_FAILED",
  1287→          targetType: "System",
  1288→          targetId: "backup",
  1289→          payload: {
  1290→            rows_verified: totalRows,
  1291→            drill_date: new Date().toISOString(),
  1292→            backup_file: backupFile,
  1293→          } as any,
  1294→        },
  1295→      });
  1296→    }
  1297→
  1298→    await endJobRun(jobRun.id, "DONE", counts);
  1299→    return { ok: true, counts };
  1300→  } catch (e: any) {
  1301→    await endJobRun(jobRun.id, "FAILED", counts, e.message);
  1302→    return { ok: false, error: e.message, counts };
  1303→  }
  1304→}
